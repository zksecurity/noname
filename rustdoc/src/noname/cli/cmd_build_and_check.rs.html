27/// Possible backends to be used in proving.

29enum BackendOpt {
30    KimchiVesta,

32    R1csBn254,

35impl From<BackendOpt> for BackendKind {
36    fn from(value: BackendOpt) -> Self {
37        match value {
38            BackendOpt::KimchiVesta => BackendKind::new_kimchi_vesta(false),
39            BackendOpt::R1csBls12_381 => BackendKind::new_r1cs_bls12_381(),
40            BackendOpt::R1csBn254 => BackendKind::new_r1cs_bn254(),
41        }
42    }
43}

81pub fn cmd_build(args: CmdBuild) -> miette::Result<()> {
82    let curr_dir = args
83        .path
84        .unwrap_or_else(|| std::env::current_dir().unwrap().try_into().unwrap());
85
86    // setup server in server mode
87    let (handle, mut server_mode) = if args.server_mode {
88        let (handle, shim) = crate::server::ServerShim::start_server();
89        (Some(handle), Some(shim))
90    } else {
91        (None, None)
92    };
93
94    // build
95    let (sources, prover_index, verifier_index) = build(
96        &curr_dir,
97        args.asm,
98        args.debug,
99        &mut server_mode,
100        args.disable_safety_check,
101    )?;
102
103    // create COMPILED_DIR
104    let compiled_path = curr_dir.join(COMPILED_DIR);
105    if compiled_path.exists() && !compiled_path.is_dir() {
106        miette::bail!("There's a filename called `{}` which collides with noname. Please delete that file first.", compiled_path);
107    }
108
109    if args.prover_params.is_none() && args.verifier_params.is_none() && !compiled_path.exists() {
110        std::fs::create_dir(&compiled_path)
111            .into_diagnostic()
112            .wrap_err(format!("could not create dir at `{compiled_path}`"))?;
113    }
114
115    // write prover
116    let prover_params = args
117        .prover_params
118        .unwrap_or(compiled_path.join("prover.nope"));
119    /*
120    std::fs::write(&prover_params, rmp_serde::to_vec(&prover_index).unwrap())
121        .into_diagnostic()
122        .wrap_err(format!(
123            "could not write prover params to `{prover_params}`"
124        ))?;
125        */
126
127    // write verifier
128    let verifier_params = args
129        .verifier_params
130        .unwrap_or(compiled_path.join("verifier.nope"));
131    std::fs::write(
132        &verifier_params,
133        rmp_serde::to_vec(&verifier_index).unwrap(),
134    )
135    .into_diagnostic()
136    .wrap_err(format!(
137        "could not write prover params to `{prover_params}`"
138    ))?;
139
140    println!("successfully built");
141
142    // server mode
143    handle.map(|h| h.join().unwrap());
144
145    //
146    Ok(())
147}

168fn add_stdlib<B: Backend>(
169    sources: &mut Sources,
170    tast: &mut TypeChecker<B>,
171    node_id: usize,
172    server_mode: &mut Option<crate::server::ServerShim>,
173) -> miette::Result<usize> {
174    let mut node_id = node_id;
175
176    // check if the release folder exists, otherwise download the latest release
177    let stdlib_dir = path_to_stdlib();
178    node_id = init_stdlib_dep(sources, tast, node_id, stdlib_dir.as_ref(), server_mode);
179
180    Ok(node_id)
181}

183fn produce_all_asts<B: Backend>(
184    path: &PathBuf,
185    server_mode: &mut Option<crate::server::ServerShim>,
186) -> miette::Result<(Sources, TypeChecker<B>)> {
187    // find manifest
188    let manifest = validate_package_and_get_manifest(&path, false)?;
189
190    // get all dependencies
191    get_deps_of_package(&manifest);
192
193    // produce dependency graph
194    let is_lib = is_lib(&path);
195
196    let this = if is_lib {
197        Some(UserRepo::new(&manifest.package.name))
198    } else {
199        None
200    };
201
202    let dep_graph = DependencyGraph::new_from_manifest(this, &manifest)?;
203
204    // produce artifacts for each dependency, starting from leaf dependencies
205    let mut sources = Sources::new();
206    let mut node_id = 0;
207
208    let mut tast = TypeChecker::new();
209
210    // adding stdlib
211    add_stdlib(&mut sources, &mut tast, node_id, server_mode)?;
212
213    for dep in dep_graph.from_leaves_to_roots() {
214        let path = path_to_package(&dep);
215
216        let lib_file = path.join("src").join("lib.no");
217        let code = std::fs::read_to_string(&lib_file)
218            .into_diagnostic()
219            .wrap_err_with(|| format!("could not read file `{path}`"))?;
220
221        let lib_file_str = lib_file.to_string();
222        node_id = typecheck_next_file(
223            &mut tast,
224            Some(dep),
225            &mut sources,
226            lib_file_str.clone(),
227            code,
228            node_id,
229            server_mode,
230        )?;
231    }
232
233    // produce artifact for this one
234    let src_dir = path.join("src");
235
236    let lib_file = src_dir.join("lib.no");
237    let main_file = src_dir.join("main.no");
238
239    // assuming that we can't have both lib.no and main.no
240    let file_path = if lib_file.exists() {
241        lib_file
242    } else {
243        main_file
244    };
245
246    let code = std::fs::read_to_string(&file_path)
247        .into_diagnostic()
248        .wrap_err_with(|| format!("could not read file `{file_path}`"))?;
249
250    let _node_id = typecheck_next_file(
251        &mut tast,
252        None,
253        &mut sources,
254        file_path.to_string(),
255        code,
256        node_id,
257        server_mode,
258    )?;
259
260    Ok((sources, tast))
261}

263pub fn build(
264    curr_dir: &PathBuf,
265    asm: bool,
266    debug: bool,
267    server_mode: &mut Option<crate::server::ServerShim>,
268    disable_safety_check: bool,
269) -> miette::Result<(Sources, ProverIndex, VerifierIndex)> {
270    // produce all TASTs
271    let (sources, tast) = produce_all_asts(curr_dir, server_mode)?;
272
273    // produce indexes
274    let double_generic_gate_optimization = false;
275
276    let kimchi_vesta = KimchiVesta::new(double_generic_gate_optimization);
277    let compiled_circuit = compile(
278        &sources,
279        tast,
280        kimchi_vesta,
281        server_mode,
282        disable_safety_check,
283    )?;
284
285    if asm {
286        println!("{}", compiled_circuit.asm(&sources, debug));
287    }
288
289    // TODO: cache artifacts
290
291    // produce indexes
292    let (prover_index, verifier_index) = compiled_circuit.compile_to_indexes()?;
293
294    Ok((sources, prover_index, verifier_index))
295}

331pub fn cmd_test(args: CmdTest) -> miette::Result<()> {
332    // parse inputs
333    let public_inputs = if let Some(s) = args.public_inputs {
334        parse_inputs(&s)?
335    } else {
336        JsonInputs::default()
337    };
338
339    let private_inputs = if let Some(s) = args.private_inputs {
340        parse_inputs(&s)?
341    } else {
342        JsonInputs::default()
343    };
344
345    match BackendKind::from(args.backend) {
346        BackendKind::KimchiVesta(_) => {
347            let (tast, sources) = typecheck_file(&args.path)?;
348            let kimchi_vesta = KimchiVesta::new(args.double);
349            let compiled_circuit = compile(
350                &sources,
351                tast,
352                kimchi_vesta,
353                &mut None,
354                args.disable_safety_check,
355            )?;
356
357            let (prover_index, verifier_index) = compiled_circuit.compile_to_indexes()?;
358            println!("successfully compiled");
359
360            // print ASM
361            let asm = prover_index.asm(&sources, args.debug);
362            println!("{asm}");
363
364            // create proof
365            let (proof, full_public_inputs, _public_output) =
366                prover_index.prove(&sources, public_inputs, private_inputs, args.debug)?;
367            println!("proof created");
368
369            // verify proof
370            verifier_index.verify(full_public_inputs, proof)?;
371            println!("proof verified");
372        }
373        BackendKind::R1csBls12_381(r1cs) => {
374            test_r1cs_backend(
375                r1cs,
376                &args.path,
377                public_inputs,
378                private_inputs,
379                args.debug,
380                args.disable_safety_check,
381            )?;
382        }
383        BackendKind::R1csBn254(r1cs) => {
384            test_r1cs_backend(
385                r1cs,
386                &args.path,
387                public_inputs,
388                private_inputs,
389                args.debug,
390                args.disable_safety_check,
391            )?;
392        }
393    }
394
395    Ok(())
396}

464fn run_r1cs_backend<F>(
465    r1cs: R1CS<F>,
466    curr_dir: &PathBuf,
467    public_inputs: JsonInputs,
468    private_inputs: JsonInputs,
469    disable_safety_check: bool,
470) -> miette::Result<()>
471where
472    F: BackendField,
473{
474    // Assuming `curr_dir`, `public_inputs`, and `private_inputs` are available in the scope
475    let (sources, tast) = produce_all_asts(curr_dir, &mut None)?;
476
477    let compiled_circuit = compile(&sources, tast, r1cs, &mut None, disable_safety_check)?;
478
479    let generated_witness =
480        generate_witness(&compiled_circuit, &sources, public_inputs, private_inputs)?;
481
482    let snarkjs_exporter = SnarkjsExporter::new(compiled_circuit.circuit.backend);
483
484    let r1cs_output_path = curr_dir.join("output.r1cs");
485    let wtns_output_path = curr_dir.join("output.wtns");
486
487    snarkjs_exporter.gen_r1cs_file(&r1cs_output_path.clone().into_string())?;
488
489    snarkjs_exporter.gen_wtns_file(&wtns_output_path.clone().into_string(), generated_witness)?;
490
491    // display the info for the generated files
492    println!("Snarkjs R1CS file generated at: {}", r1cs_output_path);
493    println!("Snarkjs Witness file generated at: {}", wtns_output_path);
494
495    Ok(())
496}

498fn test_r1cs_backend<F: BackendField>(
499    r1cs: R1CS<F>,
500    path: &PathBuf,
501    public_inputs: JsonInputs,
502    private_inputs: JsonInputs,
503    debug: bool,
504    disable_safety_check: bool,
505) -> miette::Result<()>
506where
507    F: BackendField,
508{
509    let mut sources = Sources::new();
510    let mut node_id = 0;
511
512    let mut tast = TypeChecker::new();
513
514    // adding stdlib
515    node_id = add_stdlib(&mut sources, &mut tast, node_id, &mut None)?;
516
517    let code = std::fs::read_to_string(path)
518        .into_diagnostic()
519        .wrap_err_with(|| format!("could not read file `{path}`"))?;
520
521    typecheck_next_file(
522        &mut tast,
523        None,
524        &mut sources,
525        path.to_string(),
526        code,
527        node_id,
528        &mut None,
529    )?;
530
531    let compiled_circuit = compile(&sources, tast, r1cs, &mut None, disable_safety_check)?;
532
533    generate_witness(&compiled_circuit, &sources, public_inputs, private_inputs)?;
534
535    let asm = compiled_circuit.asm(&sources, debug);
536
537    println!("{}", asm);
538
539    Ok(())
540}
