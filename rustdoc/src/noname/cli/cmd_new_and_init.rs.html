38pub fn cmd_new(args: CmdNew) -> Result<()> {
39    let path = args.path;
40
41    // for now, the package name is the same as the path
42    let package_name = path
43        .as_os_str()
44        .to_str()
45        .expect("couldn't parse the path as a string")
46        .to_owned();
47
48    if path.exists() {
49        miette::bail!("path `{path}` already exists. Use `noname init` to create a new package in an existing directory");
50    }
51
52    std::fs::create_dir(&path)
53        .into_diagnostic()
54        .wrap_err("couldn't create directory at given path")?;
55
56    mk(path, &package_name, args.lib)
57}

59pub fn cmd_init(args: CmdInit) -> Result<()> {
60    let path = args
61        .path
62        .unwrap_or(std::env::current_dir().unwrap().try_into().unwrap());
63
64    // for now, the package name is the same as the path
65    let package_name = path
66        .file_name()
67        .ok_or(miette::miette!("invalid path given in argument to CLI"))?
68        .to_string();
69
70    if !path.exists() {
71        miette::bail!("path `{path}` doesn't exists. Use `noname new` to create a new package in an non-existing directory");
72    }
73
74    if !path.is_dir() {
75        miette::bail!("path `{path}` is not a directory");
76    }
77
78    mk(path, &package_name, args.lib)
79}

81fn mk(path: PathBuf, package_name: &str, is_lib: bool) -> Result<()> {
82    let user = get_git_user()?;
83
84    let content = format!(
85        r#"[package]
86name = "{user}/{package_name}"
87version = "0.1.0"
88# see documentation at TODO for more information on how to edit this file
89
90dependencies = []
91"#
92    );
93
94    let manifest_file = path.join("Noname.toml");
95
96    if manifest_file.exists() {
97        miette::bail!("manifest file already exists at `{manifest_file}`");
98    }
99
100    std::fs::write(&manifest_file, content)
101        .into_diagnostic()
102        .wrap_err(format!(
103            "cannot create Noname.toml file at given path: `{manifest_file}`"
104        ))?;
105
106    let src_path = path.join("src");
107
108    if src_path.exists() {
109        miette::bail!("src directory already exists at `{src_path}`");
110    }
111
112    std::fs::create_dir(&src_path)
113        .into_diagnostic()
114        .wrap_err(format!(
115            "cannot create src directory at given path: `{src_path}`"
116        ))?;
117
118    let (content, file_name) = if is_lib {
119        (LIB_CONTENT, "lib.no")
120    } else {
121        (MAIN_CONTENT, "main.no")
122    };
123
124    let file_path = src_path.join(file_name);
125
126    if file_path.exists() {
127        miette::bail!("file already exists at `{file_path}`");
128    }
129
130    std::fs::write(&file_path, content)
131        .into_diagnostic()
132        .wrap_err(format!("cannot create file at given path: `{file_path}`"))?;
133
134    // success msg
135    println!("created new package at `{path}`");
136
137    Ok(())
138}

140fn get_git_user() -> Result<String> {
141    let output = std::process::Command::new("git")
142        .arg("config")
143        .arg("user.name")
144        .output()
145        .map_err(|e| MietteError::from(e))?;
146
147    if !output.status.success() {
148        let err = Error::new(ErrorKind::Other, "failed to get git username");
149        Err(MietteError::IoError(err))?
150    }
151
152    let output = String::from_utf8(output.stdout).expect("couldn't parse git output as utf8");
153
154    let username = output.trim().to_owned();
155
156    Ok(username.replace(" ", "_").to_lowercase())
157}
