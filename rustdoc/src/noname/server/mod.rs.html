26pub enum CompilerMessage {

30pub struct ServerShim {
31    tx: mpsc::Sender<CompilerMessage>,
32    rx: mpsc::Receiver<ServerMessage>,
33}
34
35impl ServerShim {
36    pub(crate) fn start_server() -> (thread::JoinHandle<()>, Self) {
37        let (compiler_to_server, rx) = mpsc::channel(1024);
38        let (tx, compiler_from_server) = mpsc::channel(1024);
39
40        // then run the server in a thread
41        let handle = thread::spawn(move || {
42            let rt = tokio::runtime::Runtime::new().unwrap();
43            rt.block_on(run_server(tx, rx));
44        });
45
46        let server_shim = ServerShim {
47            tx: compiler_to_server,
48            rx: compiler_from_server,
49        };
50
51        (handle, server_shim)
52    }
53
54    pub(crate) fn send<T: Serialize>(&self, title: String, state: &T) {
55        let state = serde_json::to_string(&state).unwrap();
56        let state = BASE64_STANDARD.encode(state);
57        let _ = self
58            .tx
59            .clone()
60            .try_send(CompilerMessage::State { title, state });
61    }
62
63    pub(crate) fn recv(&mut self) -> Option<ServerMessage> {
64        return Some(ServerMessage::Resume);
65        //self.rx.blocking_recv()
66    }
67}

73#[derive(Debug, Clone)]
74struct ServerState {
75    tx: Arc<Mutex<mpsc::Sender<ServerMessage>>>,
76    compiler_state: Arc<Mutex<Vec<CompilerStep>>>,
77}
78
79#[derive(Debug, Clone, Serialize)]
80pub struct CompilerStep {
81    pub title: String,
82    pub state: String,
83}
84
85async fn run_server(tx: mpsc::Sender<ServerMessage>, rx: mpsc::Receiver<CompilerMessage>) {
86    // initialize tracing
87    tracing_subscriber::fmt::init();
88
89    // create a new state
90    let compiler_state = Arc::new(Mutex::new(Vec::<CompilerStep>::new()));
91    let tx = Arc::new(Mutex::new(tx));
92    let rx = Arc::new(Mutex::new(rx));
93    let server_state = ServerState {
94        tx,
95        compiler_state: compiler_state.clone(),
96    };
97
98    tokio::spawn(async move {
99        while let Some(msg) = rx.lock().await.recv().await {
100            match msg {
101                CompilerMessage::State { title, state } => {
102                    let mut compiler_state = compiler_state.lock().await;
103                    compiler_state.push(CompilerStep { title, state });
104                }
105            }
106        }
107    });
108
109    // build our application with a route
110    // let manifest_dir = std::env::var("CARGO_MANIFEST_DIR").unwrap();
111    // load manifest dir from the environment, otherwise use the release directory
112    let manifest_dir =
113        std::env::var("CARGO_MANIFEST_DIR").unwrap_or_else(|_| path_to_release_dir().to_string());
114    let assets_dir = std::path::Path::new(&manifest_dir).join("assets");
115
116    let app = Router::new()
117        // let the compiler go to the next step
118        .route("/states", get(states))
119        .route("/resume/:id", get(resume))
120        // get the state at the given counter
121        .route("/state/:counter", get(get_state))
122        .nest_service("/", ServeDir::new(assets_dir))
123        .with_state(server_state);
124
125    // run our app with hyper, listening globally on port 3000
126    println!("listening on http://0.0.0.0:3000");
127    let listener = tokio::net::TcpListener::bind("0.0.0.0:3000").await.unwrap();
128    axum::serve(listener, app).await.unwrap();
129}

135#[axum::debug_handler]
136async fn states(state: State<ServerState>) -> String {
137    let compiler_state = state.compiler_state.lock().await;
138    format!("{}", compiler_state.len())
139}
140
141#[axum::debug_handler]
142async fn resume(
143    Path(id): Path<usize>,
144    state: State<ServerState>,
145) -> Result<String, (StatusCode, String)> {
146    let len = { state.compiler_state.lock().await.len() };
147    if len != id {
148        return Err((
149            StatusCode::BAD_REQUEST,
150            format!("Invalid id: {} (expected: {})", id, len),
151        ));
152    }
153    match state.tx.lock().await.send(ServerMessage::Resume).await {
154        Ok(_) => Ok("Resuming...".to_string()),
155        Err(e) => Err((
156            StatusCode::INTERNAL_SERVER_ERROR,
157            format!("Failed to send message: {}", e),
158        )),
159    }
160}

162#[derive(Clone, Debug, Serialize)]
163struct Response {
164    message: CompilerStep,
165}
166
167#[axum::debug_handler]
168async fn get_state(
169    Path(counter): Path<usize>,
170    state: State<ServerState>,
171) -> Result<Json<Response>, (StatusCode, String)> {
172    let compiler_state = state.compiler_state.lock().await;
173
174    if let Some(msg) = compiler_state.get(counter) {
175        let response = Response {
176            message: msg.clone(),
177        };
178        Ok(Json(response))
179    } else {
180        Err((StatusCode::NOT_FOUND, "State not found".to_string()))
181    }
182}
