86impl<F: BackendField> std::fmt::Debug for LinearCombination<F> {
87    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
88        // format the linear combination as a single string
89        let mut lc = String::new();
90        for (i, (var, factor)) in self.terms.iter().enumerate() {
91            if i > 0 {
92                lc.push_str(" + ");
93            }
94
95            // format the factor
96            let factor = factor.pretty();
97            match factor.as_str() {
98                "1" => lc.push_str(&format!("v_{}", var.index)),
99                _ => lc.push_str(&format!("{} * v_{}", factor, var.index)),
100            }
101        }
102        // constant
103        if self.constant != F::zero() {
104            lc.push_str(&format!(" + {}", self.constant.pretty()));
105        }
106
107        // Write the formatted string to the formatter
108        write!(f, "{}", lc)
109    }
110}

112/// Linear combination of variables and constants.
113/// For example, the linear combination is represented as a * f_a + b * f_b + f_c
114/// f_a and f_b are the coefficients of a and b respectively.
115/// a and b are represented by CellVar.
116/// The constant f_c is represented by the constant field, will always multiply with the variable at index 0 which is always 1.
117#[derive(Clone, PartialEq, Eq)]
118pub struct LinearCombination<F>
119where
120    F: BackendField,
121{
122    pub terms: HashMap<CellVar, F>,
123    pub constant: F,
124    pub span: Span,
125}
126
127impl<F> LinearCombination<F>
128where
129    F: BackendField,
130{
131    /// Convert to a CellVar.
132    /// It should
133    /// - be used when the linear combination is a single variable.
134    /// - panic if the linear combination is not a single variable or has a non-zero constant.
135    /// - panic if the single variable has factor other than 1.
136    fn to_cell_var(&self) -> &CellVar {
137        assert_eq!(self.terms.len(), 1);
138        assert_eq!(self.constant, F::zero());
139
140        let (var, factor) = self.terms.iter().next().unwrap();
141        assert_eq!(*factor, F::one());
142
143        var
144    }
145
146    /// Evaluate the linear combination with the given witness.
147    fn evaluate(&self, witness: &[F]) -> F {
148        let mut sum = F::zero();
149
150        for (var, factor) in &self.terms {
151            sum += *witness.get(var.index).unwrap() * factor;
152        }
153
154        sum += &self.constant;
155
156        sum
157    }
158
159    /// Create a linear combination to represent constant one.
160    fn one(span: Span) -> Self {
161        LinearCombination {
162            terms: HashMap::new(),
163            constant: F::one(),
164            span,
165        }
166    }
167
168    /// Create a linear combination from a constant.
169    fn from_const(cst: F, span: Span) -> Self {
170        LinearCombination {
171            terms: HashMap::new(),
172            constant: cst,
173            span,
174        }
175    }
176
177    /// Add with another linear combination.
178    fn add(&self, other: &Self, span: Span) -> Self {
179        let mut terms = self.terms.clone();
180
181        other.terms.iter().for_each(|(var, c1)| {
182            terms.entry(*var).and_modify(|c2| *c2 += c1).or_insert(*c1);
183        });
184
185        LinearCombination {
186            terms,
187            constant: self.constant + other.constant,
188            span,
189        }
190    }
191
192    /// Scale the linear combination with a constant.
193    fn scale(&self, coeff: F, span: Span) -> Self {
194        let terms = self
195            .terms
196            .iter()
197            .map(|(var, factor)| (*var, *factor * coeff))
198            .collect();
199
200        LinearCombination {
201            terms,
202            constant: self.constant * coeff,
203            span,
204        }
205    }
206
207    /// Enforces a constraint for the multiplication of two CellVars.
208    /// The constraint reduces the multiplication to a new CellVar variable,
209    /// which represents: self * other = res.
210    fn mul(&self, cs: &mut R1CS<F>, other: &Self, span: Span) -> Self {
211        let res = cs.new_internal_var(Value::Mul(self.clone(), other.clone()), span);
212        cs.enforce_constraint(self, other, &res, span);
213
214        res
215    }
216
217    /// Enforces a constraint for the equality of two CellVars.
218    /// It needs to constraint: self * 1 = other.
219    fn assert_eq(&self, cs: &mut R1CS<F>, other: &Self, span: Span) {
220        let one_cvar = LinearCombination::one(span);
221
222        cs.enforce_constraint(self, &one_cvar, other, span)
223    }
224}

239/// an R1CS constraint
240/// Each constraint comprises of 3 linear combinations from 3 matrices.
241/// It represents a constraint in math: a * b = c.
242#[derive(Clone, Debug)]
243pub struct Constraint<F>
244where
245    F: BackendField,
246{
247    pub a: LinearCombination<F>,
248    pub b: LinearCombination<F>,
249    pub c: LinearCombination<F>,
250}
251
252impl<F> Constraint<F>
253where
254    F: BackendField,
255{
256    /// Convert the 3 linear combinations to an array.
257    fn as_array(&self) -> [&LinearCombination<F>; 3] {
258        [&self.a, &self.b, &self.c]
259    }
260}

405    /// Final checks for generating the circuit.
406    /// todo: we might need to rethink about this interface
407    /// - we might just need the returned_cells argument, as the backend can record the public outputs itself?
408    fn finalize_circuit(
409        &mut self,
410        public_output: Option<crate::var::Var<Self::Field, Self::Var>>,
411        returned_cells: Option<Vec<LinearCombination<F>>>,
412        disable_safety_check: bool,
413    ) -> crate::error::Result<()> {
414        // store the return value in the public input that was created for that
415        if let Some(public_output) = public_output {
416            let cvars = &public_output.cvars;
417
418            for (pub_var, ret_var) in cvars.clone().iter().zip(returned_cells.unwrap()) {
419                // replace the computation of the public output vars with the actual variables being returned here
420                let var_idx = pub_var.cvar().unwrap().to_cell_var().index;
421                let prev = &self.witness_vector[var_idx];
422                assert!(matches!(prev, Value::PublicOutput(None)));
423                self.witness_vector[var_idx] = Value::PublicOutput(Some(ret_var));
424            }
425        }
426
427        // for sanity check, we make sure that every cellvar created has ended up in an r1cs constraint
428        let mut written_vars = HashSet::new();
429        for constraint in &self.constraints {
430            for lc in constraint.as_array() {
431                for var in lc.terms.keys() {
432                    written_vars.insert(var.index);
433                }
434            }
435        }
436
437        // check if every cell vars end up being a cell var in the circuit or public output
438        for (index, _) in self.witness_vector.iter().enumerate() {
439            // Skip the first var which is always 1
440            // - In a linear combination, each of the vars can be paired with a coefficient.
441            // - The first var is assumed to be the factor of the constant of a linear combination.
442            if index == 0 {
443                continue;
444            }
445
446            if !written_vars.contains(&index) && !disable_safety_check {
447                if let Some(private_cell_var) = self
448                    .private_input_cell_vars
449                    .iter()
450                    .find(|private_cell_var| private_cell_var.index == index)
451                {
452                    Err(Error::new(
453                        "constraint-finalization",
454                        ErrorKind::PrivateInputNotUsed,
455                        private_cell_var.span,
456                    ))?
457                } else {
458                    Err(Error::new(
459                        "constraint-finalization",
460                        ErrorKind::UnexpectedError("there's a bug in the circuit_writer, some cellvar does not end up being a cellvar in the circuit!"),
461                        Span::default(),
462                    ))?
463                }
464            }
465        }
466
467        self.finalized = true;
468
469        Ok(())
470    }

472    fn compute_var(
473        &self,
474        env: &mut crate::witness::WitnessEnv<Self::Field>,
475        lc: &LinearCombination<Self::Field>,
476    ) -> Result<Self::Field> {
477        let mut val = lc.constant;
478
479        for (var, factor) in &lc.terms {
480            let var_val = self.witness_vector.get(var.index).unwrap();
481            let calc = self.compute_val(env, var_val, var.index)? * factor;
482            val += calc;
483        }
484
485        Ok(val)
486    }

488    /// Generate the witnesses
489    /// This process should check if the constraints are satisfied.
490    fn generate_witness(
491        &self,
492        witness_env: &mut crate::witness::WitnessEnv<F>,
493        sources: &Sources,
494        typed: &Mast<Self>,
495    ) -> crate::error::Result<Self::GeneratedWitness> {
496        assert!(self.finalized, "the circuit is not finalized yet!");
497
498        // generate witness through witness vars vector
499        let mut witness = self
500            .witness_vector
501            .iter()
502            .enumerate()
503            .map(|(index, val)| {
504                match val {
505                    // Defer calculation for output vars.
506                    // The reasoning behind this is to avoid deep recursion potentially triggered by the public output var at the beginning.
507                    Value::PublicOutput(_) => Ok(F::zero()),
508                    _ => self.compute_val(witness_env, val, index),
509                }
510            })
511            .collect::<crate::error::Result<Vec<F>>>()?;
512
513        // The original vars of public outputs are not part of the constraints
514        // so we need to compute them separately
515        for var in &self.public_outputs {
516            let val = self.compute_var(witness_env, &var.to_linear_combination())?;
517            witness[var.index] = val;
518        }
519
520        self.print_log(witness_env, &self.log_info, sources, typed)?;
521
522        for (index, (constraint, debug_info)) in
523            izip!(&self.constraints, &self.debug_info).enumerate()
524        {
525            // assert a * b = c
526            let ab = constraint.a.evaluate(&witness) * constraint.b.evaluate(&witness);
527            let c = constraint.c.evaluate(&witness);
528
529            if ab != c {
530                return Err(Error::new(
531                    "runtime",
532                    ErrorKind::InvalidWitness(index),
533                    debug_info.span,
534                ));
535            }
536        }
537
538        let outputs = self
539            .public_outputs
540            .iter()
541            .map(|var| witness[var.index])
542            .collect();
543
544        Ok(GeneratedWitness { witness, outputs })
545    }

547    fn generate_asm(&self, sources: &crate::compiler::Sources, debug: bool) -> String {
548        let zero = F::zero();
549
550        let mut res = String::new();
551        res.push_str(&crate::utils::noname_version());
552
553        // add public outputs to public inputs
554        let num_public_inputs = self.public_inputs.len() + self.public_outputs.len();
555
556        // public inputs
557        res.push_str(&format!("@ public inputs: {}\n\n", num_public_inputs));
558
559        for ((row, constraint), debug_info) in
560            izip!(self.constraints.iter().enumerate(), &self.debug_info)
561        {
562            if debug {
563                // first info row: show the current row of constraints
564                res.push_str(&format!("╭{}\n", "─".repeat(80)));
565                res.push_str(&format!("│ {row} │ "));
566            }
567
568            // format the a b c linear combinations in order
569            let fmt_lcs: Vec<String> = constraint
570                .as_array()
571                .iter()
572                .map(|lc| {
573                    let mut terms: Vec<String> = lc
574                        .terms
575                        .iter()
576                        // sort by var index to make it determisitic for asm generation
577                        .sorted_by(|(a, _), (b, _)| a.index.cmp(&b.index))
578                        .map(|(var, factor)| {
579                            match factor.pretty().as_str() {
580                                // if the factor is 1, we don't need to show it
581                                "1" => format!("v_{}", var.index),
582                                _ => format!("{} * v_{}", factor.pretty(), var.index),
583                            }
584                        })
585                        .collect();
586
587                    // ignore the constant if it's zero
588                    if lc.constant != zero {
589                        terms.push(lc.constant.pretty());
590                    }
591
592                    // check if it needs to cancatenate the terms with a plus sign
593                    match terms.len() {
594                        0 => "0".to_string(),
595                        1 => terms[0].clone(),
596                        _ => terms.join(" + "),
597                    }
598                })
599                .collect();
600
601            let (a, b, c) = (&fmt_lcs[0], &fmt_lcs[1], &fmt_lcs[2]);
602
603            // format an entire constraint
604            res.push_str(&format!("{} == ({}) * ({})\n", c, a, b));
605
606            if debug {
607                // link the constraint to the source code
608                crate::utils::display_source(&mut res, sources, &[debug_info.clone()]);
609            }
610        }
611
612        res
613    }
