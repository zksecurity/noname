15#[derive(Diagnostic, Debug, Error)]
16pub enum Error {
17    /// An error type associated with [`R1CSWriter`].
18    ///
19    /// It turns out that even [`R1CSWriter`] fails to write to file,
20    /// the error is ignored and a unit type is returned,
21    /// so we come up with a custom type to represent that.
22    #[error("Something went wrong writing the R1CS file")]
23    R1CSWriterIo,
24    /// An error type associated with [`WitnessWriter`].
25    #[error(transparent)]
26    WitnessWriterIo(#[from] std::io::Error),
27}

36#[derive(Debug)]
37struct SnarkjsLinearCombination {
38    pub terms: HashMap<usize, BigInt>,
39    pub constant: BigInt,
40}
41
42impl SnarkjsLinearCombination {
43    fn try_hashmap(&self) -> Result<HashMap<usize, BigInt>, Error> {
44        let mut terms = self.terms.clone();
45
46        // add the constant term with var indexed at 0
47        if terms.insert(0, self.constant.clone()).is_some() {
48            // sanity check
49
50            Err(std::io::Error::new(
51                ErrorKind::Other,
52                "The first var should be preserved for constant term",
53            ))?
54        }
55
56        Ok(terms)
57    }
58}

60/// Calculate the number of bytes for the prime field.
61fn field_size(prime: &BigInt) -> usize {
62    if prime.bits() % 64 == 0 {
63        prime.bits() / 8
64    } else {
65        (prime.bits() / 64 + 1) * 8
66    }
67}

86    /// Restructure the linear combination to align with the snarkjs format.
87    /// - convert the factors to BigInt.
88    fn restructure_lc(&self, lc: &LinearCombination<F>) -> SnarkjsLinearCombination {
89        let terms = lc
90            .terms
91            .iter()
92            .map(|(cvar, factor)| {
93                let factor_bigint = Self::convert_to_bigint(factor);
94
95                (cvar.index, factor_bigint)
96            })
97            .collect();
98
99        let constant = Self::convert_to_bigint(&lc.constant);
100
101        SnarkjsLinearCombination { terms, constant }
102    }
103
104    /// Restructure the constraints to align with the snarkjs format.
105    fn restructure_constraints(&self) -> Vec<SnarkjsConstraint> {
106        let mut constraints = vec![];
107        for constraint in &self.r1cs_backend.constraints {
108            constraints.push(SnarkjsConstraint {
109                a: self.restructure_lc(&constraint.a),
110                b: self.restructure_lc(&constraint.b),
111                c: self.restructure_lc(&constraint.c),
112            });
113        }
114        constraints
115    }

117    /// Restructure the witness vector
118    fn restructure_witness(&self, generated_witness: GeneratedWitness<F>) -> Vec<BigInt> {
119        // convert the witness to BigInt
120        generated_witness
121            .witness
122            .iter()
123            .map(|value| Self::convert_to_bigint(value))
124            .collect()
125    }

127    /// Generate the r1cs file in snarkjs format.
128    /// It uses the circom rust library to generate the r1cs file.
129    /// The binary format spec: https://github.com/iden3/r1csfile/blob/master/doc/r1cs_bin_format.md
130    pub fn gen_r1cs_file(&self, file: &str) -> Result<(), Error> {
131        let prime = self.backend_prime();
132        let field_size = field_size(&prime);
133
134        let r1cs = R1CSWriter::new(file.to_string(), field_size, false).unwrap();
135        let mut constraint_section = R1CSWriter::start_constraints_section(r1cs).unwrap();
136
137        let restructure_constraints = self.restructure_constraints();
138
139        for constraint in &restructure_constraints {
140            ConstraintSection::write_constraint_usize(
141                &mut constraint_section,
142                &constraint.a.try_hashmap()?,
143                &constraint.b.try_hashmap()?,
144                &constraint.c.try_hashmap()?,
145            )
146            .map_err(|_| Error::R1CSWriterIo)?;
147        }
148
149        let r1cs = constraint_section.end_section().unwrap();
150        let mut header_section = R1CSWriter::start_header_section(r1cs).unwrap();
151        let header_data = HeaderData {
152            // Snarkjs uses this meta data to determine which curve to use.
153            field: prime,
154            // Both the sizes of public inputs and outputs will be crucial for the snarkjs to determine
155            // where to extract the public inputs and outputs from the witness vector.
156            // Snarkjs prove command will generate a public.json file that contains the public inputs and outputs.
157            // So if the sizes are not correct, the public.json will not be generated correctly, which might potential contains the private inputs.
158            public_outputs: self.r1cs_backend.public_outputs.len(),
159            public_inputs: self.r1cs_backend.public_inputs.len(),
160            // There seems no use of this field in the snarkjs lib. It might be just a reference.
161            private_inputs: self.r1cs_backend.private_input_number(),
162            // Add one to take into account the first var that is only added during the witness formation for snarkjs.
163            total_wires: self.r1cs_backend.witness_vector.len(),
164            // This is for circom lang debugging, so we don't need it.
165            number_of_labels: 0,
166            number_of_constraints: restructure_constraints.len(),
167        };
168        header_section
169            .write_section(header_data)
170            .map_err(|_| Error::R1CSWriterIo)?;
171
172        let r1cs = header_section.end_section().unwrap();
173        R1CSWriter::finish_writing(r1cs).map_err(|_| Error::R1CSWriterIo)
174    }
175
176    /// Generate the wtns file in snarkjs format.
177    pub fn gen_wtns_file(&self, file: &str, witness: GeneratedWitness<F>) -> Result<(), Error> {
178        let restructured_witness = self.restructure_witness(witness);
179
180        let mut witness_writer = WitnessWriter::new(file).unwrap();
181
182        Ok(witness_writer.write(restructured_witness, &self.backend_prime())?)
183    }
184
185    fn backend_prime(&self) -> BigInt {
186        BigInt::from_bytes_le(
187            num_bigint_dig::Sign::Plus,
188            &self.r1cs_backend.prime().to_bytes_le(),
189        )
190    }
191
192    fn convert_to_bigint(value: &F) -> BigInt {
193        BigInt::from_bytes_le(
194            num_bigint_dig::Sign::Plus,
195            &value.into_bigint().to_bytes_le(),
196        )
197    }

200/// Witness writer for generating the wtns file in snarkjs format.
201/// The circom rust lib seems not to have the API to generate the wtns file, so we create our own based on the snarkjs lib.
202/// The implementation follows: https://github.com/iden3/snarkjs/blob/577b3f358016a486402050d3b7242876082c085f/src/wtns_utils.js#L25
203/// It uses the same binary format as the r1cs file relies on.
204/// Although it is the same binary file protocol, but the witness file has simpler data points than the r1cs file.
205struct WitnessWriter {
206    inner: BufWriter<File>,
207    writing_section: Option<WritingSection>,
208    section_size_position: u64,
209}
210
211/// A struct to represent a section in the snarkjs binary format.
212struct WritingSection;
213
214impl WitnessWriter {
215    // Initialize a FileWriter
216    fn new(path: &str) -> Result<WitnessWriter, Error> {
217        // file type for the witness file
218        let file_type = "wtns";
219        // version of the file format
220        let version = 2u32;
221        // total number of sections
222        let n_sections = 2u32;
223
224        let file = OpenOptions::new()
225            .write(true)
226            .create(true)
227            .truncate(true)
228            .open(path)
229            .unwrap();
230        let mut writer = BufWriter::new(file);
231
232        // Write the file type (magic string) as bytes
233        let file_type_bytes = file_type.as_bytes();
234        if file_type_bytes.len() != 4 {
235            Err(std::io::Error::new(
236                ErrorKind::Other,
237                "File type must be 4 characters long",
238            ))?
239        }
240        writer.write_all(file_type_bytes)?;
241
242        // Write the version as a 32-bit unsigned integer in little endian
243        writer.write_all(&version.to_le_bytes())?;
244
245        // Write the number of sections as a 32-bit unsigned integer in little endian
246        writer.write_all(&n_sections.to_le_bytes())?;
247
248        let current_position = writer.stream_position().unwrap();
249
250        Ok(WitnessWriter {
251            inner: writer,
252            writing_section: None,
253            section_size_position: current_position,
254        })
255    }
256
257    /// Start a new section for writing.
258    fn start_write_section(&mut self, id_section: u32) -> Result<(), io::Error> {
259        // Write the section ID as ULE32
260        self.inner.write_all(&id_section.to_le_bytes())?;
261        // Get the current position
262        self.section_size_position = self.inner.stream_position().unwrap();
263        // Temporarily write 0 as ULE64 for the section size
264        self.inner.write_all(&0u64.to_le_bytes())?;
265        self.writing_section = Some(WritingSection);
266
267        Ok(())
268    }
269
270    /// End the current section
271    fn end_write_section(&mut self) -> Result<(), io::Error> {
272        let current_pos = self.inner.stream_position().unwrap();
273        // Calculate the size of the section
274        let section_size = current_pos - self.section_size_position - 8;
275
276        // Move back to where the size needs to be written
277        self.inner
278            .seek(SeekFrom::Start(self.section_size_position))?;
279        // Write the actual section size
280        self.inner.write_all(&section_size.to_le_bytes())?;
281        // Return to the end of the section
282        self.inner.seek(SeekFrom::Start(current_pos))?;
283        // Flush the buffer to ensure all data is written to the file
284        self.inner.flush()?;
285
286        self.writing_section = None;
287
288        Ok(())
289    }
290
291    /// Write the witness to the file
292    /// It stores the two sections:
293    /// - Header section: describes the field size, prime field, and the number of witnesses.
294    /// - Witness section: contains the witness values.
295    fn write(&mut self, witness: Vec<BigInt>, prime: &BigInt) -> Result<(), io::Error> {
296        // Start writing the first section
297        self.start_write_section(1)?;
298        // Write field size in number of bytes
299        let field_n_bytes = field_size(prime);
300        self.inner
301            .write_all(&(field_n_bytes as u32).to_le_bytes())?;
302        // Write the prime field in bytes
303        self.write_big_int(prime.clone(), field_n_bytes)?;
304        // Write the number of witnesses
305        self.inner
306            .write_all(&(witness.len() as u32).to_le_bytes())?;
307
308        self.end_write_section()?;
309
310        // Start writing the second section
311        self.start_write_section(2)?;
312
313        /// Write the witness values to the file
314        /// Each witness value occupies the same number of bytes as the prime field
315        for value in witness {
316            self.write_big_int(value.clone(), field_n_bytes)?;
317        }
318        self.end_write_section()
319    }
320
321    /// Write a BigInt to the file
322    fn write_big_int(&mut self, value: BigInt, size: usize) -> Result<(), io::Error> {
323        let bytes = value.to_bytes_le().1;
324
325        let mut buffer = vec![0u8; size];
326        buffer[..bytes.len()].copy_from_slice(&bytes);
327        self.inner.write_all(&buffer)
328    }
