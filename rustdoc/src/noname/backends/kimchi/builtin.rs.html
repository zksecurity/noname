20pub fn poseidon(
21    compiler: &mut CircuitWriter<KimchiVesta>,
22    _generics: &GenericParameters,
23    vars: &[VarInfo<VestaField, KimchiCellVar>],
24    span: Span,
25) -> Result<Option<Var<VestaField, KimchiCellVar>>> {
26    //
27    // sanity checks
28    //
29
30    // only one [Var] is passed
31    assert_eq!(vars.len(), 1);
32    let var_info = &vars[0];
33
34    // an array of length 2
35    match &var_info.typ {
36        Some(TyKind::Array(el_typ, 2)) => {
37            assert!(matches!(&**el_typ, TyKind::Field { .. }));
38        }
39        _ => Err(compiler.error(
40            ErrorKind::UnexpectedError(
41                "incorrect type used as input to the builtin poseidon hash function",
42            ),
43            span,
44        ))?,
45    };
46
47    // extract the values
48    let input = &var_info.var;
49    assert_eq!(input.len(), 2);
50
51    // hashing a full-constant input is not a good idea
52    if input[0].is_const() && input[1].is_const() {
53        Err(compiler.error(
54            ErrorKind::UnexpectedError("cannot hash a full-constant input"),
55            span,
56        ))?
57    }
58
59    // IMPORTANT: time to constrain any constants
60    let mut cells = vec![];
61    for const_or_cell in &input.cvars {
62        match const_or_cell {
63            ConstOrCell::Const(cst) => {
64                let cell = compiler.backend.add_constant(
65                    Some("encoding constant input to poseidon"),
66                    *cst,
67                    span,
68                );
69                cells.push(cell);
70            }
71            ConstOrCell::Cell(cell) => cells.push(*cell),
72        }
73    }
74
75    // get constants needed for poseidon
76    let poseidon_params = kimchi::mina_poseidon::pasta::fp_kimchi::params();
77
78    let rc = &poseidon_params.round_constants;
79    let width = PlonkSpongeConstantsKimchi::SPONGE_WIDTH;
80
81    // pad the input (for the capacity)
82    let zero_var = compiler.backend.add_constant(
83        Some("encoding constant 0 for the capacity of poseidon"),
84        VestaField::zero(),
85        span,
86    );
87    cells.push(zero_var);
88
89    let mut states = vec![cells.clone()];
90
91    // 0..11
92    for row in 0..POS_ROWS_PER_HASH {
93        let offset = row * ROUNDS_PER_ROW; // row * 5
94
95        // 0..5
96        for i in 0..ROUNDS_PER_ROW {
97            let mut new_state = vec![];
98
99            let prev_0 = states[states.len() - 1][0];
100            let prev_1 = states[states.len() - 1][1];
101            let prev_2 = states[states.len() - 1][2];
102
103            for col in 0..3 {
104                // create each variable
105                let var = compiler.backend.new_internal_var(
106                    Value::Hint(Arc::new(move |backend, env| {
107                        let x1 = backend.compute_var(env, &prev_0)?;
108                        let x2 = backend.compute_var(env, &prev_1)?;
109                        let x3 = backend.compute_var(env, &prev_2)?;
110
111                        let mut acc = vec![x1, x2, x3];
112
113                        // Do one full round on the previous value
114                        full_round::<VestaField, PlonkSpongeConstantsKimchi>(
115                            &kimchi::mina_poseidon::pasta::fp_kimchi::params(),
116                            &mut acc,
117                            offset + i,
118                        );
119
120                        Ok(acc[col])
121                    })),
122                    span,
123                );
124
125                new_state.push(var);
126            }
127
128            states.push(new_state);
129        }
130
131        let coeffs = (0..NUM_REGISTERS)
132            .map(|i| rc[offset + (i / width)][i % width])
133            .collect();
134
135        let vars = vec![
136            Some(states[offset][0]),
137            Some(states[offset][1]),
138            Some(states[offset][2]),
139            Some(states[offset + 4][0]),
140            Some(states[offset + 4][1]),
141            Some(states[offset + 4][2]),
142            Some(states[offset + 1][0]),
143            Some(states[offset + 1][1]),
144            Some(states[offset + 1][2]),
145            Some(states[offset + 2][0]),
146            Some(states[offset + 2][1]),
147            Some(states[offset + 2][2]),
148            Some(states[offset + 3][0]),
149            Some(states[offset + 3][1]),
150            Some(states[offset + 3][2]),
151        ];
152
153        compiler.backend.add_gate(
154            "uses a poseidon gate to constrain 5 rounds of poseidon",
155            GateKind::Poseidon,
156            vars,
157            coeffs,
158            span,
159        );
160    }
161
162    let final_state = &states[states.len() - 1];
163    let final_row = vec![
164        Some(final_state[0]),
165        Some(final_state[1]),
166        Some(final_state[2]),
167    ];
168
169    // zero gate to store the result
170    compiler.backend.add_gate(
171        "uses a zero gate to store the output of poseidon",
172        GateKind::Zero,
173        final_row.clone(),
174        vec![],
175        span,
176    );
177
178    let vars = final_row
179        .iter()
180        .flatten()
181        .cloned()
182        .map(ConstOrCell::Cell)
183        .collect();
184
185    Ok(Some(Var::new(vars, span)))
186}
