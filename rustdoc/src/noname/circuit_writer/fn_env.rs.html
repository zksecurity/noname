9/// Information about a variable.
10#[derive(Debug, Clone)]
11pub struct VarInfo<F, C>
12where
13    F: BackendField,
14    C: BackendVar,
15{
16    /// The variable.
17    pub var: Var<F, C>,
18
19    // TODO: we could also store this on the expression node... but I think this is lighter
20    pub mutable: bool,
21
22    /// We keep track of the type of variables, eventhough we're not in the typechecker anymore,
23    /// because we need to know the type for method calls.
24    // TODO: why is this an option?
25    pub typ: Option<TyKind>,
26}
27
28impl<F: BackendField, C: BackendVar> VarInfo<F, C> {
29    pub fn new(var: Var<F, C>, mutable: bool, typ: Option<TyKind>) -> Self {
30        Self { var, mutable, typ }
31    }
32
33    pub fn reassign(&self, var: Var<F, C>) -> Self {
34        Self {
35            var,
36            mutable: self.mutable,
37            typ: self.typ.clone(),
38        }
39    }
40
41    pub fn reassign_range(&self, var: Var<F, C>, start: usize, len: usize) -> Self {
42        // sanity check
43        assert_eq!(var.len(), len);
44
45        // create new cvars by modifying a specific range
46        let mut cvars = self.var.cvars.clone();
47        let cvars_range = &mut cvars[start..start + len];
48        cvars_range.clone_from_slice(&var.cvars);
49
50        let var = Var::new(cvars, self.var.span);
51
52        Self {
53            var,
54            mutable: self.mutable,
55            typ: self.typ.clone(),
56        }
57    }
58}

60/// Is used to store functions' scoped variables.
61/// This include inputs and output of the function,  as well as local variables.
62/// You can think of it as a function call stack.
63#[derive(Default, Debug, Clone)]
64pub struct FnEnv<F, C>
65where
66    F: BackendField,
67    C: BackendVar,
68{
69    /// The current nesting level.
70    /// Starting at 0 (top level), and increasing as we go into a block.
71    current_scope: usize,
72
73    /// Used by the private and public inputs,
74    /// and any other external variables created in the circuit
75    /// This needs to be garbage collected when we exit a scope.
76    /// Note: The `usize` is the scope in which the variable was created.
77    vars: HashMap<String, (usize, VarInfo<F, C>)>,
78}
79
80impl<F: BackendField, C: BackendVar> FnEnv<F, C> {
81    /// Creates a new FnEnv
82    pub fn new() -> Self {
83        Self {
84            current_scope: 0,
85            vars: HashMap::new(),
86        }
87    }
88
89    /// Enters a scoped block.
90    pub fn nest(&mut self) {
91        self.current_scope += 1;
92    }
93
94    /// Exits a scoped block.
95    pub fn pop(&mut self) {
96        self.current_scope = self.current_scope.checked_sub(1).expect("scope bug");
97
98        // remove variables as we exit the scope
99        // (we don't need to keep them around to detect shadowing,
100        // as we already did that in type checker)
101        let mut to_delete = vec![];
102        for (name, (scope, _)) in self.vars.iter() {
103            if *scope > self.current_scope {
104                to_delete.push(name.clone());
105            }
106        }
107        for d in to_delete {
108            self.vars.remove(&d);
109        }
110    }
111
112    /// Returns true if a scope is a prefix of our scope.
113    fn is_in_scope(&self, prefix_scope: usize) -> bool {
114        self.current_scope >= prefix_scope
115    }
116
117    /// Stores type information about a local variable.
118    /// Note that we forbid shadowing at all scopes.
119    pub fn add_local_var(&mut self, var_name: String, var_info: VarInfo<F, C>) {
120        let scope = self.current_scope;
121
122        if self
123            .vars
124            .insert(var_name.clone(), (scope, var_info))
125            .is_some()
126        {
127            panic!("type checker error: var `{var_name}` already exists");
128        }
129    }
130
131    /// Retrieves type information on a variable, given a name.
132    /// If the variable is not in scope, return false.
133    // TODO: return an error no?
134    pub fn get_local_var(&self, var_name: &str) -> VarInfo<F, C> {
135        let (scope, var_info) = self
136            .vars
137            .get(var_name)
138            .unwrap_or_else(|| panic!("type checking bug: local variable `{var_name}` not found"));
139        if !self.is_in_scope(*scope) {
140            panic!("type checking bug: local variable `{var_name}` not in scope");
141        }
142
143        var_info.clone()
144    }
145
146    pub fn reassign_local_var(&mut self, var_name: &str, var: Var<F, C>) {
147        // get the scope first, we don't want to modify that
148        let (scope, var_info) = self
149            .vars
150            .get(var_name)
151            .expect("type checking bug: local variable for reassigning not found");
152
153        if !self.is_in_scope(*scope) {
154            panic!("type checking bug: local variable for reassigning not in scope");
155        }
156
157        if !var_info.mutable {
158            panic!("type checking bug: local variable for reassigning is not mutable");
159        }
160
161        let var_info = var_info.reassign(var);
162        self.vars.insert(var_name.to_string(), (*scope, var_info));
163    }
164
165    /// Same as [Self::reassign_var], but only reassigns a specific range of the variable.
166    pub fn reassign_var_range(&mut self, var_name: &str, var: Var<F, C>, start: usize, len: usize) {
167        // get the scope first, we don't want to modify that
168        let (scope, var_info) = self
169            .vars
170            .get(var_name)
171            .expect("type checking bug: local variable for reassigning not found");
172
173        if !self.is_in_scope(*scope) {
174            panic!("type checking bug: local variable for reassigning not in scope");
175        }
176
177        if !var_info.mutable {
178            panic!("type checking bug: local variable for reassigning is not mutable");
179        }
180
181        let var_info = var_info.reassign_range(var, start, len);
182        self.vars.insert(var_name.to_string(), (*scope, var_info));
183    }
184}
