26#[derive(Debug)]
27pub struct CircuitWriter<B>
28where
29    B: Backend,
30{
31    /// The monomorphized state for the main module.
32    // The process walks through the monomorphized AST to generate the circuit.
33    typed: Mast<B>,
34
35    /// The constraint backend for the circuit.
36    /// For now, this needs to be exposed for the kimchi prover for kimchi specific low level data.
37    /// So we might make this private if the prover facilities can be deprecated.
38    pub backend: B,
39
40    /// If a public output is set, this will be used to store its [Var].
41    /// The public output generation works as follows:
42    /// 1. This cvar is created and inserted in the circuit (gates) during compilation of the public input
43    ///    (as the public output is the end of the public input)
44    /// 2. When the `return` statement of the circuit is parsed,
45    ///    it will set this `public_output` variable again to the correct vars.
46    /// 3. During witness generation, the public output computation
47    ///    is delayed until the very end.
48    pub(crate) public_output: Option<Var<B::Field, B::Var>>,
49
50    ir_writer: ir::IRWriter<B>,
51}

84    pub fn add_local_var(
85        &self,
86        fn_env: &mut FnEnv<B::Field, B::Var>,
87        var_name: String,
88        var_info: VarInfo<B::Field, B::Var>,
89    ) -> Result<()> {
90        // check for consts first
91        let qualified = FullyQualified::local(var_name.clone());
92        if let Some(_cst_info) = self.typed.const_info(&qualified) {
93            Err(Error::new("add-local-var", ErrorKind::UnexpectedError("type checker bug: we already have a constant with the same name (`{var_name}`)!"), Span::default()))?
94        }
95        Ok(fn_env.add_local_var(var_name, var_info))
96    }
97
98    pub fn get_local_var(
99        &self,
100        fn_env: &FnEnv<B::Field, B::Var>,
101        var_name: &str,
102    ) -> VarInfo<B::Field, B::Var> {
103        // check for consts first
104        let qualified = FullyQualified::local(var_name.to_string());
105        if let Some(cst_info) = self.typed.const_info(&qualified) {
106            let var = Var::new_constant_typ(cst_info, cst_info.typ.span);
107            return VarInfo::new(var, false, Some(TyKind::Field { constant: true }));
108        }
109
110        // then check for local variables
111        fn_env.get_local_var(var_name)
112    }

114    /// Retrieves the [FnInfo] for the `main()` function.

142    pub fn generate_circuit(
143        typed: Mast<B>,
144        backend: B,
145        disable_safety_check: bool,
146    ) -> Result<CompiledCircuit<B>> {
147        // create circuit writer
148        let mut circuit_writer = CircuitWriter::new(typed, backend);
149
150        // get main function
151        let qualified = FullyQualified::local("main".to_string());
152        let main_fn_info = circuit_writer.main_info()?;
153
154        let function = match &main_fn_info.kind {
155            crate::imports::FnKind::BuiltIn(_, _, _) => unreachable!(),
156            crate::imports::FnKind::Native(fn_sig) => fn_sig.clone(),
157        };
158
159        // initialize the circuit
160        circuit_writer.backend.init_circuit();
161
162        // create the main env
163        let fn_env = &mut FnEnv::new();
164
165        // create public output
166        if let Some(typ) = &function.sig.return_type {
167            // whatever is the size of return type, we need to add that many public outputs
168            let size_of = circuit_writer.size_of(&typ.kind);
169            circuit_writer.add_public_outputs(size_of, typ.span);
170        }
171
172        // public inputs should be handled first
173        for arg in function.sig.arguments.iter().filter(|arg| arg.is_public()) {
174            match &arg.attribute {
175                Some(attr) => {
176                    if !matches!(attr.kind, AttributeKind::Pub) {
177                        return Err(
178                            circuit_writer.error(ErrorKind::InvalidAttribute(attr.kind), attr.span)
179                        );
180                    }
181                }
182                None => Err(Error::new(
183                    "generate-circuit",
184                    ErrorKind::UnexpectedError("public arguments must have a pub attribute"),
185                    Span::default(),
186                ))?,
187            }
188            circuit_writer.handle_arg(arg, fn_env, CircuitWriter::add_public_inputs)?;
189        }
190
191        // then handle private inputs
192        for arg in function.sig.arguments.iter().filter(|arg| !arg.is_public()) {
193            circuit_writer.handle_arg(arg, fn_env, CircuitWriter::add_private_inputs)?;
194        }
195
196        // compile function
197        let returned_cells = circuit_writer.compile_main_function(fn_env, &function)?;
198        let main_span = circuit_writer.main_info().unwrap().span;
199        let public_output = circuit_writer.public_output.clone();
200
201        // constraint public outputs to the result of the circuit
202        if let Some(public_output) = &public_output {
203            let span = match circuit_writer.main_info().as_ref().unwrap() {
204                FnInfo {
205                    kind: crate::imports::FnKind::Native(fn_def),
206                    ..
207                } => fn_def.body.last().unwrap().span,
208                _ => unreachable!(),
209            };
210            let cvars = &public_output.cvars;
211
212            for (pub_var, ret_var) in cvars.iter().zip(&returned_cells.clone().unwrap()) {
213                circuit_writer
214                    .backend
215                    .assert_eq_var(pub_var.cvar().unwrap(), ret_var, span);
216            }
217        }
218
219        circuit_writer.backend.finalize_circuit(
220            public_output,
221            returned_cells,
222            disable_safety_check,
223        )?;
224
225        //
226        Ok(CompiledCircuit::new(circuit_writer))
227    }

229    /// A wrapper for the backend generate_witness
230    pub fn generate_witness(
231        &self,
232        witness_env: &mut WitnessEnv<B::Field>,

237    }
238
239    fn handle_arg(
240        &mut self,
241        arg: &FnArg,
242        fn_env: &mut FnEnv<B::Field, B::Var>,
243        handle_input: fn(&mut CircuitWriter<B>, String, usize, Span) -> Var<B::Field, B::Var>,
244    ) -> Result<()> {
245        let FnArg { name, typ, .. } = arg;
246
247        // get length
248        let len = self.size_of(&typ.kind);
249
250        // create the variable
251        let var = handle_input(self, name.value.clone(), len, name.span);
252
253        // constrain what needs to be constrained
254        // (for example, booleans need to be constrained to be 0 or 1)
255        // note: we constrain private inputs as well as public inputs
256        // in theory we might not need to check the validity of public inputs,
257        // but we are being extra cautious due to attacks
258        // where the prover gives the verifier malformed inputs that look legit.
259        // (See short address attacks in Ethereum.)
260        self.constrain_inputs_to_main(&var.cvars, &typ.kind, typ.span)?;
261
262        // add argument variable to the ast env
263        let mutable = false; // TODO: should we add a mut keyword in arguments as well?
264        let var_info = VarInfo::new(var, mutable, Some(typ.kind.clone()));
265        self.add_local_var(fn_env, name.value.clone(), var_info)?;
266
267        Ok(())
268    }
