116    fn compile_stmt(
117        &mut self,
118        fn_env: &mut FnEnv<B::Field, B::Var>,
119        stmt: &Stmt,
120    ) -> Result<Option<VarOrRef<B>>> {
121        match &stmt.kind {
122            StmtKind::Assign { mutable, lhs, rhs } => {
123                // compute the rhs
124                let rhs_var = self
125                    .compute_expr(fn_env, rhs)?
126                    .ok_or_else(|| self.error(ErrorKind::CannotComputeExpression, stmt.span))?;
127
128                // obtain the actual values
129                let rhs_var = rhs_var.value(self, fn_env);
130
131                let typ = self.expr_type(rhs).cloned();
132                let var_info = VarInfo::new(rhs_var, *mutable, typ);
133
134                // store the new variable
135                // TODO: do we really need to store that in the scope? That's not an actual var in the scope that's an internal var...
136                self.add_local_var(fn_env, lhs.value.clone(), var_info)?;
137            }
138
139            StmtKind::ForLoop {
140                var,
141                argument,
142                body,
143            } => {
144                match argument {
145                    ForLoopArgument::Range(range) => {
146                        // compute the start and end of the range
147                        let start_bg: BigUint = self
148                            .compute_expr(fn_env, &range.start)?
149                            .ok_or_else(|| {
150                                self.error(ErrorKind::CannotComputeExpression, range.start.span)
151                            })?
152                            .constant()
153                            .expect("expected constant")
154                            .into();
155                        let start: u32 = start_bg.try_into().map_err(|_| {
156                            self.error(ErrorKind::InvalidRangeSize, range.start.span)
157                        })?;
158
159                        let end_bg: BigUint = self
160                            .compute_expr(fn_env, &range.end)?
161                            .ok_or_else(|| {
162                                self.error(ErrorKind::CannotComputeExpression, range.end.span)
163                            })?
164                            .constant()
165                            .expect("expected constant")
166                            .into();
167                        let end: u32 = end_bg
168                            .try_into()
169                            .map_err(|_| self.error(ErrorKind::InvalidRangeSize, range.end.span))?;
170
171                        // compute for the for loop block
172                        for ii in start..end {
173                            fn_env.nest();
174
175                            let cst_var = Var::new_constant(ii.into(), var.span);
176                            let var_info = VarInfo::new(
177                                cst_var,
178                                false,
179                                Some(TyKind::Field { constant: true }),
180                            );
181                            self.add_local_var(fn_env, var.value.clone(), var_info)?;
182
183                            self.compile_block(fn_env, body)?;
184
185                            fn_env.pop();
186                        }
187                    }
188                    ForLoopArgument::Iterator(iterator) => {
189                        let iterator_var = self
190                            .compute_expr(fn_env, iterator)?
191                            .expect("array access on non-array");
192
193                        let array_typ = self
194                            .expr_type(iterator)
195                            .cloned()
196                            .expect("cannot find type of array");
197
198                        let (elem_type, array_len) = match array_typ {
199                            TyKind::Array(ty, array_len) => (ty, array_len),
200                            _ => Err(Error::new(
201                                "compile-stmt",
202                                ErrorKind::UnexpectedError("expected array"),
203                                stmt.span,
204                            ))?,
205                        };
206
207                        // compute the size of each element in the array
208                        let len = self.size_of(&elem_type);
209
210                        for idx in 0..array_len {
211                            // compute the real index
212                            let idx = idx as usize;
213                            let start = idx * len;
214
215                            fn_env.nest();
216
217                            // add the variable to the inner environment corresponding
218                            // to iterator[idx]
219                            let indexed_var = iterator_var.narrow(start, len).value(self, fn_env);
220                            let var_info =
221                                VarInfo::new(indexed_var.clone(), false, Some(*elem_type.clone()));
222                            self.add_local_var(fn_env, var.value.clone(), var_info)?;
223
224                            self.compile_block(fn_env, body)?;
225
226                            fn_env.pop();
227                        }
228                    }
229                }
230            }
231            StmtKind::Expr(expr) => {
232                // compute the expression
233                let var = self.compute_expr(fn_env, expr)?;
234
235                // make sure it does not return any value.
236                assert!(var.is_none());
237            }
238            StmtKind::Return(expr) => {
239                let var = self
240                    .compute_expr(fn_env, expr)?
241                    .ok_or_else(|| self.error(ErrorKind::CannotComputeExpression, stmt.span))?;
242
243                // we already checked in type checking that this is not an early return
244                return Ok(Some(var));
245            }
246            StmtKind::Comment(_) => (),
247        }
248
249        Ok(None)
250    }

252    /// might return something?
253    fn compile_block(
254        &mut self,
255        fn_env: &mut FnEnv<B::Field, B::Var>,
256        stmts: &[Stmt],
257    ) -> Result<Option<Var<B::Field, B::Var>>> {
258        fn_env.nest();
259        for stmt in stmts {
260            let res = self.compile_stmt(fn_env, stmt)?;
261            if let Some(var) = res {
262                // a block doesn't return a pointer, only values
263                let var = var.value(self, fn_env);
264
265                // we already checked for early returns in type checking
266                return Ok(Some(var));
267            }
268        }
269        fn_env.pop();
270        Ok(None)
271    }

273    fn compile_native_function_call(
274        &mut self,
275        function: &FunctionDef,
276        args: Vec<VarInfo<B::Field, B::Var>>,
277    ) -> Result<Option<Var<B::Field, B::Var>>> {
278        assert!(!function.is_main());
279
280        // create new fn_env
281        let fn_env = &mut FnEnv::new();
282
283        // set arguments
284        assert_eq!(function.sig.arguments.len(), args.len());
285
286        for (name, var_info) in function.sig.arguments.iter().zip(args) {
287            self.add_local_var(fn_env, name.name.value.clone(), var_info)?;
288        }
289
290        // compile it and potentially return a return value
291        self.compile_block(fn_env, &function.body)
292    }

294    pub(crate) fn constrain_inputs_to_main(
295        &mut self,
296        input: &[ConstOrCell<B::Field, B::Var>],
297        input_typ: &TyKind,
298        span: Span,
299    ) -> Result<()> {
300        match input_typ {
301            TyKind::Field { constant: false } => (),
302            TyKind::Bool => {
303                assert_eq!(input.len(), 1);
304                boolean::check(self, &input[0], span);
305            }
306            TyKind::Array(tykind, _) => {
307                let el_size = self.size_of(tykind);
308                for el in input.chunks(el_size) {
309                    self.constrain_inputs_to_main(el, tykind, span)?;
310                }
311            }
312            TyKind::Custom {
313                module,
314                name: struct_name,
315            } => {
316                let qualified = FullyQualified::new(module, &struct_name);
317                let struct_info = self
318                    .struct_info(&qualified)
319                    .ok_or(self.error(ErrorKind::UnexpectedError("struct not found"), span))?
320                    .clone();
321
322                let mut offset = 0;
323                for (_field_name, field_typ, _attribute) in &struct_info.fields {
324                    let len = self.size_of(field_typ);
325                    let range = offset..(offset + len);
326                    self.constrain_inputs_to_main(&input[range], field_typ, span)?;
327                    offset += len;
328                }
329            }
330            TyKind::Field { constant: true } => unreachable!(),
331            TyKind::GenericSizedArray(_, _) => {
332                unreachable!("generic array should have been resolved")
333            }
334            TyKind::String(_) => todo!("String type is not supported for constraints"),
335            TyKind::Tuple(types) => {
336                let mut offset = 0;
337                for ty in types {
338                    let size = self.size_of(ty);
339                    let slice = &input[offset..(offset + size)];
340                    self.constrain_inputs_to_main(slice, input_typ, span)?;
341                    offset += size;
342                }
343            }
344        };
345        Ok(())
346    }

348    /// Compile a function. Used to compile `main()` only for now
349    pub(crate) fn compile_main_function(
350        &mut self,
351        fn_env: &mut FnEnv<B::Field, B::Var>,
352        function: &FunctionDef,
353    ) -> Result<Option<Vec<B::Var>>> {
354        assert!(function.is_main());
355
356        // compile the block
357        let returned = self.compile_block(fn_env, &function.body)?;
358
359        // we're expecting something returned?
360        match (function.sig.return_type.as_ref(), returned) {
361            (None, None) => Ok(None),
362            (Some(expected), None) => Err(self.error(ErrorKind::MissingReturn, expected.span)),
363            (None, Some(returned)) => Err(self.error(ErrorKind::UnexpectedReturn, returned.span)),
364            (Some(_expected), Some(returned)) => {
365                // make sure there are no constants in the returned value
366                let mut returned_cells = vec![];
367                for r in &returned.cvars {
368                    match r {
369                        ConstOrCell::Cell(c) => returned_cells.push(c.clone()),
370                        ConstOrCell::Const(_) => {
371                            Err(self.error(ErrorKind::ConstantInOutput, returned.span))?
372                        }
373                    }
374                }
375
376                self.public_output
377                    .as_ref()
378                    .expect("bug in the compiler: missing public output");
379
380                Ok(Some(returned_cells))
381            }
382        }
383    }

385    fn compute_expr(
386        &mut self,
387        fn_env: &mut FnEnv<B::Field, B::Var>,
388        expr: &Expr,
389    ) -> Result<Option<VarOrRef<B>>> {
390        match &expr.kind {
391            // `module::fn_name(args)`
392            ExprKind::FnCall {
393                module,
394                fn_name,
395                args,
396                ..
397            } => {
398                // sanity check
399                if fn_name.value == "main" {
400                    Err(self.error(ErrorKind::RecursiveMain, expr.span))?
401                }
402
403                // retrieve the function in the env
404                let qualified = FullyQualified::new(module, &fn_name.value);
405                let fn_info = self
406                    .fn_info(&qualified)
407                    .ok_or_else(|| {
408                        self.error(
409                            ErrorKind::UndefinedFunction(fn_name.value.clone()),
410                            fn_name.span,
411                        )
412                    })?
413                    .clone();
414
415                // compute the arguments
416                // module::fn_name(args)
417                //                 ^^^^
418                let mut vars = Vec::with_capacity(args.len());
419                for arg in args {
420                    // get the variable behind the expression
421                    let var = self
422                        .compute_expr(fn_env, arg)?
423                        .ok_or_else(|| self.error(ErrorKind::CannotComputeExpression, arg.span))?;
424
425                    // we pass variables by values always
426                    let var = var.value(self, fn_env);
427
428                    let typ = self.expr_type(arg).cloned();
429                    let mutable = false; // TODO: mut keyword in arguments?
430                    let var_info = VarInfo::new(var, mutable, typ);
431
432                    vars.push(var_info);
433                }
434
435                match &fn_info.kind {
436                    // assert() <-- for example
437                    FnKind::BuiltIn(sig, handle, _) => {
438                        let res = handle(self, &sig.generics, &vars, expr.span);
439                        res.map(|r| r.map(VarOrRef::Var))
440                    }
441
442                    // fn_name(args)
443                    // ^^^^^^^
444                    FnKind::Native(func) => {
445                        // module::fn_name(args)
446                        // ^^^^^^
447                        if func.is_hint {
448                            self.ir_writer
449                                .compile_hint_function_call(func, vars)
450                                .map(|r| {
451                                    let cvars: Vec<_> = r
452                                        .into_iter()
453                                        .map(|r| {
454                                            ConstOrCell::Cell(
455                                                self.backend.new_internal_var(r, expr.span),
456                                            )
457                                        })
458                                        .collect();
459
460                                    if cvars.is_empty() {
461                                        return None;
462                                    }
463
464                                    Some(VarOrRef::Var(Var::new(cvars, expr.span)))
465                                })
466                        } else {
467                            self.compile_native_function_call(func, vars)
468                                .map(|r| r.map(VarOrRef::Var))
469                        }
470                    }
471                }
472            }
473
474            ExprKind::FieldAccess { lhs, rhs } => {
475                // get var behind lhs
476                let lhs_var = self
477                    .compute_expr(fn_env, lhs)?
478                    .ok_or_else(|| self.error(ErrorKind::CannotComputeExpression, lhs.span))?;
479
480                // get struct info behind lhs
481                let lhs_struct = self
482                    .expr_type(lhs)
483                    .ok_or_else(|| self.error(ErrorKind::CannotComputeExpression, lhs.span))?;
484
485                let (module, self_struct) = match lhs_struct {
486                    TyKind::Custom { module, name } => (module, name),
487                    _ => Err(Error::new(
488                        "compute-expr",
489                        ErrorKind::UnexpectedError(
490                            "could not figure out struct implementing that method call",
491                        ),
492                        lhs.span,
493                    ))?,
494                };
495
496                let qualified = FullyQualified::new(module, self_struct);
497                let struct_info = self
498                    .struct_info(&qualified)
499                    .expect("struct info not found for custom struct");
500
501                // find range of field
502                let mut start = 0;
503                let mut len = 0;
504                for (field, field_typ, _attribute) in &struct_info.fields {
505                    if field == &rhs.value {
506                        len = self.size_of(field_typ);
507                        break;
508                    }
509
510                    start += self.size_of(field_typ);
511                }
512
513                // narrow the variable to the given range
514                let var = lhs_var.narrow(start, len);
515                Ok(Some(var))
516            }
517
518            // `Thing.method(args)` or `thing.method(args)`
519            ExprKind::MethodCall {
520                lhs,
521                method_name,
522                args,
523            } => {
524                // figure out the name of the custom struct
525                let lhs_typ = self.expr_type(lhs).expect("method call on what?").clone();
526
527                let (module, struct_name) = match &lhs_typ {
528                    TyKind::Custom { module, name } => (module, name),
529                    _ => {
530                        return Err(self.error(
531                            ErrorKind::UnexpectedError("method call only work on custom types"),
532                            lhs.span,
533                        ))
534                    }
535                };
536
537                // get var of `self`
538                // (might be `None` if it's a static method call)
539                let self_var = self.compute_expr(fn_env, lhs)?;
540
541                // find method info
542                let qualified = FullyQualified::new(module, struct_name);
543                let struct_info = self
544                    .struct_info(&qualified)
545                    .ok_or(self.error(
546                        ErrorKind::UnexpectedError("struct not found"),
547                        method_name.span,
548                    ))?
549                    .clone();
550                let func = struct_info
551                    .methods
552                    .get(&method_name.value)
553                    .expect("could not find method");
554
555                // if method has a `self` argument, manually add it to the list of argument
556                let mut vars = vec![];
557                if let Some(first_arg) = func.sig.arguments.first() {
558                    if first_arg.name.value == "self" {
559                        let self_var = self_var.ok_or_else(|| {
560                            self.error(ErrorKind::NotAStaticMethod, method_name.span)
561                        })?;
562
563                        // TODO: for now we pass `self` by value as well
564                        let mutable = false;
565                        let self_var = self_var.value(self, fn_env);
566
567                        let self_var_info = VarInfo::new(self_var, mutable, Some(lhs_typ.clone()));
568                        vars.insert(0, self_var_info);
569                    }
570                } else {
571                    assert!(self_var.is_none());
572                }
573
574                // compute the arguments
575                for arg in args {
576                    let var = self
577                        .compute_expr(fn_env, arg)?
578                        .ok_or_else(|| self.error(ErrorKind::CannotComputeExpression, arg.span))?;
579
580                    // TODO: for now we pass `self` by value as well
581                    let mutable = false;
582                    let var = var.value(self, fn_env);
583
584                    let typ = self.expr_type(arg).cloned();
585                    let var_info = VarInfo::new(var, mutable, typ);
586
587                    vars.push(var_info);
588                }
589
590                // execute method
591                self.compile_native_function_call(func, vars)
592                    .map(|r| r.map(VarOrRef::Var))
593            }
594
595            ExprKind::IfElse { cond, then_, else_ } => {
596                let cond = self
597                    .compute_expr(fn_env, cond)?
598                    .unwrap()
599                    .value(self, fn_env);
600                let then_ = self
601                    .compute_expr(fn_env, then_)?
602                    .unwrap()
603                    .value(self, fn_env);
604                let else_ = self
605                    .compute_expr(fn_env, else_)?
606                    .unwrap()
607                    .value(self, fn_env);
608
609                let res = field::if_else(self, &cond, &then_, &else_, expr.span);
610
611                Ok(Some(VarOrRef::Var(res)))
612            }
613
614            ExprKind::Assignment { lhs, rhs } => {
615                // figure out the local var  of lhs
616                let lhs = self.compute_expr(fn_env, lhs)?.unwrap();
617
618                // figure out the var of what's on the right
619                let rhs = self.compute_expr(fn_env, rhs)?.unwrap();
620                let rhs_var = match rhs {
621                    VarOrRef::Var(var) => var,
622                    VarOrRef::Ref {
623                        var_name,
624                        start,
625                        len,
626                    } => {
627                        let var_info = self.get_local_var(fn_env, &var_name);
628                        let cvars = var_info.var.range(start, len).to_vec();
629                        Var::new(cvars, var_info.var.span)
630                    }
631                };
632
633                // replace the left with the right
634                match lhs {
635                    VarOrRef::Var(_) => Err(Error::new(
636                        "compute-expr",
637                        ErrorKind::UnexpectedError("can't reassign this non-mutable variable"),
638                        expr.span,
639                    ))?,
640                    VarOrRef::Ref {
641                        var_name,
642                        start,
643                        len,
644                    } => {
645                        fn_env.reassign_var_range(&var_name, rhs_var, start, len);
646                    }
647                }
648
649                Ok(None)
650            }
651
652            ExprKind::BinaryOp { op, lhs, rhs, .. } => {
653                let lhs = self.compute_expr(fn_env, lhs)?.unwrap();
654                let rhs = self.compute_expr(fn_env, rhs)?.unwrap();
655
656                let lhs = lhs.value(self, fn_env);
657                let rhs = rhs.value(self, fn_env);
658
659                let res = match op {
660                    Op2::Addition => field::add(self, &lhs[0], &rhs[0], expr.span),
661                    Op2::Subtraction => field::sub(self, &lhs[0], &rhs[0], expr.span),
662                    Op2::Multiplication => field::mul(self, &lhs[0], &rhs[0], expr.span),
663                    Op2::Equality => field::equal(self, &lhs, &rhs, expr.span),
664                    Op2::Inequality => field::not_equal(self, &lhs, &rhs, expr.span),
665                    Op2::BoolAnd => boolean::and(self, &lhs[0], &rhs[0], expr.span),
666                    Op2::BoolOr => boolean::or(self, &lhs[0], &rhs[0], expr.span),
667                    Op2::Division => todo!(),
668                };
669
670                Ok(Some(VarOrRef::Var(res)))
671            }
672
673            ExprKind::Negated(b) => {
674                let var = self.compute_expr(fn_env, b)?.unwrap();
675
676                let var = var.value(self, fn_env);
677
678                todo!()
679            }
680
681            ExprKind::Not(b) => {
682                let var = self.compute_expr(fn_env, b)?.unwrap();
683
684                let var = var.value(self, fn_env);
685
686                let res = boolean::not(self, &var[0], expr.span.merge_with(b.span));
687                Ok(Some(VarOrRef::Var(res)))
688            }
689
690            ExprKind::BigUInt(b) => {
691                let ff = B::Field::try_from(b.to_owned()).map_err(|_| {
692                    self.error(ErrorKind::CannotConvertToField(b.to_string()), expr.span)
693                })?;
694
695                let res = VarOrRef::Var(Var::new_constant(ff, expr.span));
696                Ok(Some(res))
697            }
698
699            ExprKind::Bool(b) => {
700                let value = if *b {
701                    B::Field::one()
702                } else {
703                    B::Field::zero()
704                };
705                let res = VarOrRef::Var(Var::new_constant(value, expr.span));
706                Ok(Some(res))
707            }
708
709            ExprKind::StringLiteral(s) => {
710                let chars_in_ff: Vec<B::Field> =
711                    s.chars().map(|char| B::Field::from(char as u8)).collect();
712                let cvars = chars_in_ff.iter().map(|&f| ConstOrCell::Const(f)).collect();
713                Ok(Some(VarOrRef::Var(Var::new(cvars, expr.span))))
714            }
715
716            ExprKind::Variable { module, name } => {
717                // if it's a type we return nothing
718                // (most likely what follows is a static method call)
719                if is_type(&name.value) {
720                    return Ok(None);
721                }
722
723                // search for constants first
724                let qualified = FullyQualified::new(module, &name.value);
725                let var_info = if let Some(cst_info) = self.const_info(&qualified) {
726                    let var = Var::new_constant_typ(cst_info, name.span);
727                    VarInfo::new(var, false, Some(cst_info.typ.kind.clone()))
728                } else {
729                    // if no constant found, look in the function's scope
730                    // remember: we can do this because the type checker already checked that we didn't shadow constant vars
731                    self.get_local_var(fn_env, &name.value)
732                };
733
734                let res = VarOrRef::from_var_info(name.value.clone(), var_info);
735                Ok(Some(res))
736            }
737
738            ExprKind::ArrayOrTupleAccess { container, idx } => {
739                // retrieve var of container
740                let var = self
741                    .compute_expr(fn_env, container)?
742                    .expect("container access on non-container");
743
744                // compute the index
745                let idx_var = self
746                    .compute_expr(fn_env, idx)?
747                    .ok_or_else(|| self.error(ErrorKind::CannotComputeExpression, expr.span))?;
748                let idx = idx_var
749                    .constant()
750                    .ok_or_else(|| self.error(ErrorKind::ExpectedConstant, expr.span))?;
751                let idx: BigUint = idx.into();
752                let idx: usize = idx.try_into().unwrap();
753
754                // retrieve the type of the elements in the container
755                let container_typ = self
756                    .expr_type(container)
757                    .expect("cannot find type of container");
758
759                // actual starting index for narrowing the var depends on the cotainer
760                // for arrays it is just idx * elem_size as all elements are of same size
761                // while for tuples we have to sum the sizes of all types up to that index
762                let (start, len) = match container_typ {
763                    TyKind::Array(ty, array_len) => {
764                        if idx >= (*array_len as usize) {
765                            return Err(self.error(
766                                ErrorKind::ArrayIndexOutOfBounds(idx, *array_len as usize - 1),
767                                expr.span,
768                            ));
769                        }
770                        let len = self.size_of(ty);
771                        let start = idx * self.size_of(ty);
772                        (start, len)
773                    }
774
775                    TyKind::Tuple(typs) => {
776                        let mut start = 0;
777                        for i in 0..idx {
778                            start += self.size_of(&typs[i]);
779                        }
780                        (start, self.size_of(&typs[idx]))
781                    }
782                    _ => Err(Error::new(
783                        "compute-expr",
784                        ErrorKind::UnexpectedError("expected container"),
785                        expr.span,
786                    ))?,
787                };
788
789                // out-of-bound checks
790                if start >= var.len() || start + len > var.len() {
791                    return Err(self.error(
792                        ErrorKind::ArrayIndexOutOfBounds(start, var.len()),
793                        expr.span,
794                    ));
795                }
796
797                // index into the var
798                let var = var.narrow(start, len);
799
800                //
801                Ok(Some(var))
802            }
803
804            ExprKind::ArrayDeclaration(items) => {
805                let mut cvars = vec![];
806
807                for item in items {
808                    let var = self.compute_expr(fn_env, item)?.unwrap();
809                    let to_extend = var.value(self, fn_env).cvars.clone();
810                    cvars.extend(to_extend);
811                }
812
813                let var = VarOrRef::Var(Var::new(cvars, expr.span));
814
815                Ok(Some(var))
816            }
817
818            ExprKind::CustomTypeDeclaration { custom: _, fields } => {
819                // create the struct by just concatenating all of its cvars
820                let mut cvars = vec![];
821                for (_field, rhs) in fields {
822                    let var = self.compute_expr(fn_env, rhs)?.unwrap();
823                    let to_extend = var.value(self, fn_env).cvars.clone();
824                    cvars.extend(to_extend);
825                }
826                let var = VarOrRef::Var(Var::new(cvars, expr.span));
827
828                //
829                Ok(Some(var))
830            }
831            ExprKind::RepeatedArrayInit { item, size } => {
832                let size = self
833                    .compute_expr(fn_env, size)?
834                    .ok_or_else(|| self.error(ErrorKind::CannotComputeExpression, expr.span))?;
835                let size = size
836                    .constant()
837                    .ok_or_else(|| self.error(ErrorKind::ExpectedConstant, expr.span))?;
838                let size: BigUint = size.into();
839                let size: usize = size.try_into().unwrap();
840
841                let mut cvars = vec![];
842                for _ in 0..size {
843                    let var = self.compute_expr(fn_env, item)?.unwrap();
844                    let to_extend = var.value(self, fn_env).cvars.clone();
845                    cvars.extend(to_extend);
846                }
847
848                let var = VarOrRef::Var(Var::new(cvars, expr.span));
849                Ok(Some(var))
850            }
851            // exact copy of Array Declaration there is nothing really different at when looking it from a expression level
852            // as both of them are just `Vec<Expr>`
853            ExprKind::TupleDeclaration(items) => {
854                let mut cvars = vec![];
855
856                for item in items {
857                    let var = self.compute_expr(fn_env, item)?.unwrap();
858                    let to_extend = var.value(self, fn_env).cvars.clone();
859                    cvars.extend(to_extend);
860                }
861
862                let var = VarOrRef::Var(Var::new(cvars, expr.span));
863
864                Ok(Some(var))
865            }
866        }
867    }

869    pub fn add_public_inputs(
870        &mut self,
871        name: String,
872        num: usize,
873        span: Span,
874    ) -> Var<B::Field, B::Var> {
875        let mut cvars = Vec::with_capacity(num);
876
877        for idx in 0..num {
878            let cvar = self
879                .backend
880                .add_public_input(Value::External(name.clone(), idx), span);
881            cvars.push(ConstOrCell::Cell(cvar));
882        }
883
884        Var::new(cvars, span)
885    }
886
887    pub fn add_public_outputs(&mut self, num: usize, span: Span) {
888        assert!(self.public_output.is_none());
889
890        let mut cvars = Vec::with_capacity(num);
891        for _ in 0..num {
892            let cvar = self
893                .backend
894                .add_public_output(Value::PublicOutput(None), span);
895            cvars.push(ConstOrCell::Cell(cvar));
896        }
897
898        // store it
899        let res = Var::new(cvars, span);
900        self.public_output = Some(res);
901    }
902
903    pub fn add_private_inputs(
904        &mut self,
905        name: String,
906        num: usize,
907        span: Span,
908    ) -> Var<B::Field, B::Var> {
909        let mut cvars = Vec::with_capacity(num);
910
911        for idx in 0..num {
912            // create the var
913            let cvar = self
914                .backend
915                .add_private_input(Value::External(name.clone(), idx), span);
916            cvars.push(ConstOrCell::Cell(cvar));
917        }
918
919        Var::new(cvars, span)
920    }
