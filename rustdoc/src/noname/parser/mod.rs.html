50/// A context for the parser.
51#[derive(Debug, Default)]
52pub struct ParserCtx {
53    /// A counter used to uniquely identify different nodes in the AST.
54    pub node_id: usize,
55
56    /// Used mainly for error reporting,
57    /// when we don't have a token to read
58    // TODO: replace with `(usize::MAX, usize::MAX)`?
59    pub last_token: Option<Token>,
60
61    /// The file we're parsing
62    pub filename_id: usize,
63}
64
65impl ParserCtx {
66    pub fn new(filename_id: usize, node_id: usize) -> Self {
67        Self {
68            node_id,
69            last_token: None,
70            filename_id,
71        }
72    }
73
74    pub fn error(&self, kind: ErrorKind, span: Span) -> Error {
75        Error::new("parser", kind, span)
76    }
77
78    /// Returns a new unique node id.
79    pub fn next_node_id(&mut self) -> usize {
80        self.node_id += 1;
81        self.node_id
82    }
83
84    // TODO: I think I don't need this, I should always be able to use the last token I read if I don't see anything, otherwise maybe just write -1 to say "EOF"
85    pub fn last_span(&self) -> Span {
86        let span = self
87            .last_token
88            .as_ref()
89            .map(|token| token.span)
90            .unwrap_or(Span::new(self.filename_id, 0, 0));
91        Span::new(self.filename_id, span.end(), 0)
92    }
93}

99#[derive(Debug, Default, Serialize)]
100#[serde(bound(serialize = "Root<B::Field>: Serialize",))]
101pub struct AST<B: Backend>(pub Vec<Root<B::Field>>);
102
103impl<B: Backend> AST<B> {
104    pub fn parse(
105        filename_id: usize,
106        mut tokens: Tokens,
107        node_id: usize,
108    ) -> Result<(AST<B>, usize)> {
109        let mut ast = vec![];
110        let ctx = &mut ParserCtx::new(filename_id, node_id);
111
112        // use statements must appear first
113        let mut function_observed = false;
114
115        while let Some(token) = tokens.bump(ctx) {
116            match &token.kind {
117                // `use crypto::poseidon;`
118                TokenKind::Keyword(Keyword::Use) => {
119                    if function_observed {
120                        return Err(ctx.error(ErrorKind::UseAfterFn, token.span));
121                    }
122
123                    let path = UsePath::parse(ctx, &mut tokens)?;
124                    ast.push(Root {
125                        kind: RootKind::Use(path),
126                        span: token.span,
127                    });
128
129                    // end of line
130                    let next_token = tokens.bump(ctx);
131                    if !matches!(
132                        next_token,
133                        Some(Token {
134                            kind: TokenKind::SemiColon,
135                            ..
136                        })
137                    ) {
138                        return Err(ctx.error(ErrorKind::InvalidEndOfLine, token.span));
139                    }
140                }
141
142                // `const FOO = 42;`
143                TokenKind::Keyword(Keyword::Const) => {
144                    let cst = ConstDef::parse(ctx, &mut tokens)?;
145
146                    ast.push(Root {
147                        kind: RootKind::ConstDef(cst),
148                        span: token.span,
149                    });
150                }
151
152                // `fn main() { }`
153                TokenKind::Keyword(Keyword::Fn) => {
154                    function_observed = true;
155
156                    let func = FunctionDef::parse(ctx, &mut tokens)?;
157                    ast.push(Root {
158                        kind: RootKind::FunctionDef(func),
159                        span: token.span,
160                    });
161                }
162
163                // `hint fn calc() { }`
164                TokenKind::Keyword(Keyword::Hint) => {
165                    // expect fn token
166                    tokens.bump_expected(ctx, TokenKind::Keyword(Keyword::Fn))?;
167
168                    function_observed = true;
169
170                    let func = FunctionDef::parse_hint(ctx, &mut tokens)?;
171
172                    ast.push(Root {
173                        kind: RootKind::FunctionDef(func),
174                        span: token.span,
175                    });
176                }
177
178                // `struct Foo { a: Field, b: Field }`
179                TokenKind::Keyword(Keyword::Struct) => {
180                    let s = StructDef::parse(ctx, &mut tokens)?;
181                    ast.push(Root {
182                        kind: RootKind::StructDef(s),
183                        span: token.span,
184                    });
185                }
186
187                // `// some comment`
188                TokenKind::Comment(comment) => {
189                    ast.push(Root {
190                        kind: RootKind::Comment(comment.clone()),
191                        span: token.span,
192                    });
193                }
194
195                // unrecognized
196                _ => {
197                    return Err(ctx.error(ErrorKind::InvalidToken, token.span));
198                }
199            }
200        }
201
202        Ok((Self(ast), ctx.node_id))
203    }
204}
