26pub fn parse_type_declaration(
27    ctx: &mut ParserCtx,
28    tokens: &mut Tokens,
29    ident: Ident,
30) -> Result<Expr> {
31    if !is_type(&ident.value) {
32        return Err(ctx.error(
33            ErrorKind::UnexpectedError(
34                "this looks like a type declaration but not on a type (types start with an uppercase)",
35            ), ident.span));
36    }
37
38    let mut span = ident.span;
39
40    // Thing { x: 1, y: 2 }
41    //       ^
42    tokens.bump(ctx);
43
44    let mut fields = vec![];
45
46    // Thing { x: 1, y: 2 }
47    //         ^^^^^^^^^^^^
48    loop {
49        // Thing { x: 1, y: 2 }
50        //                    ^
51        if let Some(Token {
52            kind: TokenKind::RightCurlyBracket,
53            ..
54        }) = tokens.peek()
55        {
56            tokens.bump(ctx);
57            break;
58        };
59
60        // Thing { x: 1, y: 2 }
61        //         ^
62        let field_name = Ident::parse(ctx, tokens)?;
63
64        // Thing { x: 1, y: 2 }
65        //          ^
66        tokens.bump_expected(ctx, TokenKind::Colon)?;
67
68        // Thing { x: 1, y: 2 }
69        //            ^
70        let field_value = Expr::parse(ctx, tokens)?;
71        span = span.merge_with(field_value.span);
72        fields.push((field_name, field_value));
73
74        // Thing { x: 1, y: 2 }
75        //             ^      ^
76        match tokens.bump_err(ctx, ErrorKind::InvalidEndOfLine)? {
77            Token {
78                kind: TokenKind::Comma,
79                ..
80            } => (),
81            Token {
82                kind: TokenKind::RightCurlyBracket,
83                ..
84            } => break,
85            _ => return Err(ctx.error(ErrorKind::InvalidEndOfLine, ctx.last_span())),
86        };
87    }
88
89    Ok(Expr::new(
90        ctx,
91        ExprKind::CustomTypeDeclaration {
92            custom: CustomType {
93                module: ModulePath::Local,
94                name: ident.value,
95                span: ident.span,
96            },
97            fields,
98        },
99        span,
100    ))
101}

103pub fn parse_fn_call_args(ctx: &mut ParserCtx, tokens: &mut Tokens) -> Result<(Vec<Expr>, Span)> {
104    let start = tokens.bump(ctx).expect("parser error: parse_fn_call_args"); // (
105    let mut span = start.span;
106
107    let mut args = vec![];
108    loop {
109        let pp = tokens.peek();
110
111        match pp {
112            Some(x) => match x.kind {
113                // ,
114                TokenKind::Comma => {
115                    tokens.bump(ctx);
116                }
117
118                // )
119                TokenKind::RightParen => {
120                    let end = tokens.bump(ctx).unwrap();
121                    span = span.merge_with(end.span);
122                    break;
123                }
124
125                // an argument (as expression)
126                _ => {
127                    let arg = Expr::parse(ctx, tokens)?;
128
129                    args.push(arg);
130                }
131            },
132
133            None => {
134                return Err(ctx.error(
135                    ErrorKind::InvalidFnCall("unexpected end of function call"),
136                    ctx.last_span(),
137                ))
138            }
139        }
140    }
141
142    Ok((args, span))
143}

183pub enum Symbolic {

185    Concrete(BigUint),
186    /// Point to a constant variable

189    Generic(Ident),

195impl Display for Symbolic {

197        match self {

201            Symbolic::Add(lhs, rhs) => write!(f, "{} + {}", lhs, rhs),
202            Symbolic::Sub(lhs, rhs) => write!(f, "{} - {}", lhs, rhs),

208impl Symbolic {
209    /// Extract all generic parameters.
210    /// Since the function signature syntax doesn't support <N, M> to declare generics,
211    /// we need to extract the implicit generic parameters from the function arguments.
212    /// Then they will be attached to [FnSig]
213    pub fn extract_generics(&self) -> HashSet<String> {
214        let mut generics = HashSet::new();
215
216        match self {
217            Symbolic::Concrete(_) => (),
218            Symbolic::Constant(ident) => {
219                generics.insert(ident.value.clone());
220            }
221            Symbolic::Generic(ident) => {
222                generics.insert(ident.value.clone());
223            }
224            Symbolic::Add(lhs, rhs) | Symbolic::Mul(lhs, rhs) | Symbolic::Sub(lhs, rhs) => {
225                generics.extend(lhs.extract_generics());
226                generics.extend(rhs.extract_generics());
227            }
228        }
229
230        generics
231    }
232
233    /// Parse from an expression node recursively.
234    pub fn parse(node: &Expr) -> Result<Self> {
235        match &node.kind {
236            ExprKind::BigUInt(n) => Ok(Symbolic::Concrete(n.clone())),
237            ExprKind::Variable { module: _, name } => {
238                if is_generic_parameter(&name.value) {
239                    Ok(Symbolic::Generic(name.clone()))
240                } else {
241                    Ok(Symbolic::Constant(name.clone()))
242                }
243            }
244            ExprKind::BinaryOp {
245                op,
246                lhs,
247                rhs,
248                protected: _,
249            } => {
250                let lhs = Symbolic::parse(lhs)?;
251                let rhs = Symbolic::parse(rhs);
252
253                // no protected flags are needed, as this is based on expression nodes which already ordered the operations
254                match op {
255                    Op2::Addition => Ok(Symbolic::Add(Box::new(lhs), Box::new(rhs?))),
256                    Op2::Subtraction => Ok(Symbolic::Sub(Box::new(lhs), Box::new(rhs?))),
257                    Op2::Multiplication => Ok(Symbolic::Mul(Box::new(lhs), Box::new(rhs?))),
258                    _ => Err(Error::new(
259                        "mast",
260                        ErrorKind::InvalidSymbolicSize,
261                        node.span,
262                    )),
263                }
264            }
265            _ => Err(Error::new(
266                "mast",
267                ErrorKind::InvalidSymbolicSize,
268                node.span,
269            )),
270        }
271    }
272}

285    /// A boolean (`true` or `false`).
286    Bool,

289    Tuple(Vec<TyKind>),

291    // U8,
292    // U16,
293    // U32,
294    // U64,
295    /// An array with symbolic size.
296    /// This is an intermediate type.
297    /// After monomorphized, it will be converted to `Array`.
298    GenericSizedArray(Box<TyKind>, Symbolic),
299
300    /// A string  type current purpose it to pass around for logging
301    String(String),

304impl TyKind {
305    /// Compares two `TyKind` types to determine if they match based on the `no_generic_allowed` flag.
306    ///
307    /// # Parameters
308    /// - `expected`: The expected `TyKind` type to compare against.
309    /// - `no_generic_allowed`: A flag indicating whether to disallow matching with generic types.
310    ///
311    /// # Returns
312    /// - `true` if the `self` and `expected` types match according to the rules defined and the `no_generic_allowed` flag.
313    /// - `false` otherwise.
314    ///
315    /// # Matching Rules
316    /// - If both types are `Field`, the function considers them a match.
317    /// - For `Array` types, it checks if both the size and element types match recursively.
318    /// - For `GenericSizedArray` types:
319    ///   - If `no_generic_allowed` is `true`, the function returns `false`.
320    ///   - If `no_generic_allowed` is `false`, the function compares the element types.
321    /// - For `Custom` types, it compares the `module` and `name` fields for equality.
322    /// - For tuples, it matches type of each element i.e `self[i] == expected[i]` for every i
323    /// - For other types, it uses a basic equality check.
324    pub fn match_expected(&self, expected: &TyKind, no_generic_allowed: bool) -> bool {
325        match (self, expected) {
326            (TyKind::Field { .. }, TyKind::Field { .. }) => true,
327            (TyKind::Array(lhs, lhs_size), TyKind::Array(rhs, rhs_size)) => {
328                lhs_size == rhs_size && lhs.match_expected(rhs, no_generic_allowed)
329            }
330            (TyKind::GenericSizedArray(lhs, _), TyKind::GenericSizedArray(rhs, _))
331            | (TyKind::Array(lhs, _), TyKind::GenericSizedArray(rhs, _))
332            | (TyKind::GenericSizedArray(lhs, _), TyKind::Array(rhs, _)) => {
333                if no_generic_allowed {
334                    false
335                } else {
336                    lhs.match_expected(rhs, no_generic_allowed)
337                }
338            }
339            (
340                TyKind::Custom { module, name },
341                TyKind::Custom {
342                    module: expected_module,
343                    name: expected_name,
344                },
345            ) => module == expected_module && name == expected_name,
346            (TyKind::String { .. }, TyKind::String { .. }) => true,
347            (x, y) if x == y => true,
348            (TyKind::Tuple(lhs), TyKind::Tuple(rhs)) => {
349                // if length does not match then they are of different type
350                if lhs.len() == rhs.len() {
351                    let match_items = lhs
352                        .iter()
353                        .zip(rhs)
354                        .filter(|&(l, r)| l.match_expected(r, no_generic_allowed))
355                        .count();
356                    lhs.len() == match_items
357                } else {
358                    false
359                }
360            }
361            _ => false,
362        }
363    }
364    /// Recursively extract generic parameters from GenericArray type
365    /// it should be able to extract generic parameter 'N' 'M' from [[Field; N], M]
366    pub fn extract_generics(&self) -> HashSet<String> {
367        let mut generics = HashSet::new();
368
369        match self {
370            TyKind::Field { .. } => (),
371            TyKind::Bool => (),
372            TyKind::Custom { .. } => (),
373            // e.g [[Field; N], 3]
374            TyKind::Array(ty, _) => {
375                generics.extend(ty.extract_generics());
376            }
377            // e.g [[Field; N], M]
378            TyKind::GenericSizedArray(ty, sym) => {
379                generics.extend(ty.extract_generics());
380                generics.extend(sym.extract_generics());
381            }
382            TyKind::String { .. } => (),
383            // for the time when (([Field;N]))
384            TyKind::Tuple(typs) => {
385                for ty in typs {
386                    generics.extend(ty.extract_generics());
387                }
388            }
389        }
390
391        generics
392    }
393}

444impl Ty {
445    pub fn reserved_types(module: ModulePath, name: Ident) -> Result<TyKind> {
446        let reserved_type = match name.value.as_ref() {
447            "Field" | "Bool" if !matches!(module, ModulePath::Local) => Err(Error::new(
448                "reserved_types",
449                ErrorKind::UnexpectedError("reserved types cannot be in a module"),
450                name.span,
451            ))?,
452            // Default the `constant` to false, as here has no context for const attribute.
453            // For a function argument and it is with const attribute,
454            // the `constant` will be corrected to true by the `FunctionDef::parse_args` parser.
455            "Field" => TyKind::Field { constant: false },
456            "Bool" => TyKind::Bool,
457            _ => TyKind::Custom {
458                module,
459                name: name.value,
460            },
461        };
462        Ok(reserved_type)
463    }
464
465    pub fn parse(ctx: &mut ParserCtx, tokens: &mut Tokens) -> Result<Self> {
466        let token = tokens.bump_err(ctx, ErrorKind::MissingType)?;
467
468        match token.kind {
469            // module::Type or Type
470            // ^^^^^^^^^^^^    ^^^^
471            TokenKind::Identifier(ty_name) => {
472                let maybe_module = Ident::new(ty_name.clone(), token.span);
473                let (module, name, _span) = if is_type(&ty_name) {
474                    // Type
475                    // ^^^^
476                    (ModulePath::Local, maybe_module, token.span)
477                } else {
478                    // module::Type
479                    //       ^^
480                    tokens.bump_expected(ctx, TokenKind::DoubleColon)?;
481
482                    // module::Type
483                    //         ^^^^
484                    let (name, span) = match tokens.bump(ctx) {
485                        Some(Token {
486                            kind: TokenKind::Identifier(name),
487                            span,
488                        }) => (name, span),
489                        _ => return Err(ctx.error(ErrorKind::MissingType, ctx.last_span())),
490                    };
491
492                    let name = Ident::new(name, span);
493                    let span = token.span.merge_with(span);
494
495                    (ModulePath::Alias(maybe_module), name, span)
496                };
497
498                let ty_kind = Self::reserved_types(module, name)?;
499
500                Ok(Self {
501                    kind: ty_kind,
502                    span: token.span,
503                })
504            }
505
506            // array
507            // [type; size]
508            // ^
509            TokenKind::LeftBracket => {
510                let span = token.span;
511
512                // [type; size]
513                //   ^
514                let ty = Ty::parse(ctx, tokens)?;
515
516                // [type; size]
517                //      ^
518                tokens.bump_expected(ctx, TokenKind::SemiColon)?;
519
520                // [type; size]
521                //         ^
522                let expr = Expr::parse(ctx, tokens)?;
523                match expr.kind {
524                    ExprKind::BigUInt(n) => {
525                        tokens.bump_expected(ctx, TokenKind::RightBracket)?;
526                        Ok(Ty {
527                            kind: TyKind::Array(Box::new(ty.kind), n.try_into().unwrap()),
528                            span,
529                        })
530                    }
531                    _ => {
532                        tokens.bump_expected(ctx, TokenKind::RightBracket)?;
533                        Ok(Ty {
534                            kind: TyKind::GenericSizedArray(
535                                Box::new(ty.kind),
536                                Symbolic::parse(&expr)?,
537                            ),
538                            span,
539                        })
540                    }
541                }
542            }
543
544            // tuple type return
545            TokenKind::LeftParen => {
546                let mut typs = vec![];
547                loop {
548                    // parse the type
549                    let ty = Ty::parse(ctx, tokens)?;
550                    typs.push(ty.kind);
551
552                    // if next token is RightParen then return the type
553                    let token = tokens.peek();
554                    match token {
555                        Some(token) => match token.kind {
556                            TokenKind::RightParen => {
557                                tokens.bump(ctx);
558                                return Ok(Ty {
559                                    kind: TyKind::Tuple(typs),
560                                    span: token.span,
561                                });
562                            }
563                            // if there is a comma just bump the tokens so we are on the type
564                            TokenKind::Comma => _ = tokens.bump(ctx),
565                            _ => return Err(ctx.error(ErrorKind::InvalidEndOfLine, token.span)),
566                        },
567                        _ => (),
568                    }
569                }
570            }
571
572            // unrecognized
573            _ => Err(ctx.error(ErrorKind::InvalidType, token.span)),
574        }
575    }
576}

581//~ Backus–Naur Form (BNF) grammar:
582//~
583//~ fn_sig ::= ident "(" param { "," param } ")" [ return_val ]
584//~ return_val ::= "->" type

588impl FnSig {
589    pub fn parse(ctx: &mut ParserCtx, tokens: &mut Tokens) -> Result<Self> {
590        let (name, kind) = FuncOrMethod::parse(ctx, tokens)?;
591
592        let arguments = FunctionDef::parse_args(ctx, tokens, &kind)?;
593
594        // extract generic parameters from arguments
595        let mut generics = GenericParameters::default();
596        for arg in &arguments {
597            match &arg.typ.kind {
598                TyKind::Field { .. } => {
599                    // extract from const argument
600                    if is_generic_parameter(&arg.name.value) && arg.is_constant() {
601                        generics.add(arg.name.value.to_string());
602                    }
603                }
604                TyKind::Array(ty, _) => {
605                    // recursively extract all generic parameters from the item type
606                    let extracted = ty.extract_generics();
607
608                    for name in extracted {
609                        generics.add(name);
610                    }
611                }
612                TyKind::GenericSizedArray(_, _) => {
613                    // recursively extract all generic parameters from the symbolic size
614                    let extracted = arg.typ.kind.extract_generics();
615
616                    for name in extracted {
617                        generics.add(name);
618                    }
619                }
620                // extracts generics from interior of tuple
621                TyKind::Tuple(typs) => {
622                    for ty in typs {
623                        let extracted = ty.extract_generics();
624                        for name in extracted {
625                            generics.add(name);
626                        }
627                    }
628                }
629                _ => (),
630            }
631        }
632
633        let return_type = FunctionDef::parse_fn_return_type(ctx, tokens)?;
634
635        Ok(Self {
636            kind,
637            name,
638            generics,
639            arguments,
640            return_type,
641        })
642    }

644    /// Recursively assign values to the generic parameters based on observed Array type argument
645    pub fn resolve_generic_array(
646        &mut self,
647        sig_arg: &TyKind,
648        observed: &TyKind,
649        span: Span,
650    ) -> Result<()> {
651        match (sig_arg, observed) {
652            // [[Field; NN]; MM]
653            (TyKind::GenericSizedArray(ty, sym), TyKind::Array(observed_ty, observed_size)) => {
654                // resolve the generic parameter
655                match sym {
656                    Symbolic::Generic(ident) => {
657                        self.generics.assign(
658                            &ident.value,
659                            BigUint::from_u32(*observed_size).unwrap(),
660                            span,
661                        )?;
662                    }
663                    _ => unreachable!("no operation allowed on symbolic size in function argument"),
664                }
665
666                // recursively resolve the generic parameter
667                self.resolve_generic_array(ty, observed_ty, span)?;
668            }
669            // [[Field; NN]; 3]
670            (TyKind::Array(ty, _), TyKind::Array(observed_ty, _)) => {
671                // recursively resolve the generic parameter
672                self.resolve_generic_array(ty, observed_ty, span)?;
673            }
674            _ => (),
675        }
676
677        Ok(())
678    }

680    /// Returns true if the function signature contains generic parameters or generic array types.
681    /// Either:
682    /// - `const NN: Field` or `[[Field; NN]; MM]`
683    /// - `[Field; cst]`, where cst is a constant variable. We also monomorphize generic array with a constant var as its size.
684    /// - `([Field; cst])` when tuple type returns a generic
685    pub fn require_monomorphization(&self) -> bool {
686        let has_arg_cst = self
687            .arguments
688            .iter()
689            .any(|arg| self.has_constant(&arg.typ.kind));
690
691        let has_ret_cst = self.return_type.is_some()
692            && self.has_constant(&self.return_type.as_ref().unwrap().kind);
693
694        !self.generics.is_empty() || has_arg_cst || has_ret_cst
695    }

697    /// Recursively check if the generic array symbolic value contains constant variant
698    fn has_constant(&self, typ: &TyKind) -> bool {
699        match typ {
700            TyKind::GenericSizedArray(ty, sym) => {
701                match sym {
702                    Symbolic::Constant(_) => return true,
703                    _ => false,
704                };
705
706                self.has_constant(ty)
707            }
708            TyKind::Array(ty, _) => self.has_constant(ty),
709            TyKind::Tuple(typs) => typs.iter().any(|ty| self.has_constant(ty)),
710            _ => false,
711        }
712    }

714    /// Returns the monomorphized function name,
715    /// using the patter: `fn_full_qualified_name#generic1=value1#generic2=value2`
716    pub fn monomorphized_name(&self) -> Ident {
717        let mut name = self.name.clone();
718
719        // check if it contains # in the name
720        if name.value.contains('#') {
721            // if so, then it is already monomorphized
722            return name;
723        }
724
725        if self.require_monomorphization() {
726            let mut generics = self.generics.parameters.iter().collect::<Vec<_>>();
727            generics.sort_by(|a, b| a.0.cmp(b.0));
728
729            let generics = generics
730                .iter()
731                .map(|(name, value)| format!("{}={}", name, value.as_ref().unwrap()))
732                .collect::<Vec<_>>()
733                .join("#");
734
735            name.value.push_str(&format!("#{}", generics));
736        }
737
738        name
739    }
740}

851impl GenericParameters {
852    /// Return all generic parameter names
853    pub fn names(&self) -> HashSet<String> {
854        self.parameters.keys().cloned().collect()
855    }

857    /// Add an unbound generic parameter
858    pub fn add(&mut self, name: String) {
859        self.parameters.insert(name, None);
860    }

863    pub fn get(&self, name: &str) -> BigUint {
864        self.parameters
865            .get(name)
866            .expect("generic parameter not found")
867            .as_ref()
868            .expect("generic value not assigned")
869            .clone()
870    }
871
872    /// Returns whether the generic parameters are empty
873    pub fn is_empty(&self) -> bool {
874        self.parameters.is_empty()
875    }
876
877    /// Bind a generic parameter to a value
878    pub fn assign(&mut self, name: &String, value: BigUint, span: Span) -> Result<()> {
879        let existing = self.parameters.get(name);
880        match existing {
881            Some(Some(v)) => {
882                if *v == value {
883                    return Ok(());
884                }
885
886                Err(Error::new(
887                    "mast",
888                    ErrorKind::ConflictGenericValue(
889                        name.to_string(),
890                        v.to_str_radix(10),
891                        value.to_str_radix(10),
892                    ),
893                    span,
894                ))
895            }
896            Some(None) => {
897                self.parameters.insert(name.to_string(), Some(value));
898                Ok(())
899            }
900            None => Err(Error::new(
901                "mast",
902                ErrorKind::UnexpectedGenericParameter(name.to_string()),
903                span,
904            )),
905        }
906    }

909        self.resolved_sig = Some(ResolvedSig {
910            arguments,
911            return_type,
912        });
913    }

964    pub fn extract_generic_names(&self) -> HashSet<String> {
965        let mut generics = HashSet::new();
966
967        match &self.typ.kind {
968            TyKind::Field { .. } => {
969                // extract from const argument
970                if is_generic_parameter(&self.name.value) && self.is_constant() {
971                    generics.insert(self.name.value.to_string());
972                }
973            }
974            TyKind::Array(ty, _) => {
975                // recursively extract all generic parameters from the item type
976                let extracted = ty.extract_generics();
977
978                for name in extracted {
979                    generics.insert(name);
980                }
981            }
982            TyKind::GenericSizedArray(_, _) => {
983                // recursively extract all generic parameters from the symbolic size
984                let extracted = self.typ.kind.extract_generics();
985
986                for name in extracted {
987                    generics.insert(name);
988                }
989            }
990            // extract generics for inner type
991            TyKind::Tuple(typs) => {
992                for ty in typs {
993                    let extracted = self.typ.kind.extract_generics();
994                    for name in extracted {
995                        generics.insert(name);
996                    }
997                }
998            }
999            _ => (),
1000        }
1001
1002        generics
1003    }

1006impl FuncOrMethod {
1007    pub fn parse(ctx: &mut ParserCtx, tokens: &mut Tokens) -> Result<(Ident, Self)> {
1008        // fn House.verify(   or   fn verify(
1009        //    ^^^^^                   ^^^^^
1010        let maybe_self_name = tokens.bump_ident(
1011            ctx,
1012            ErrorKind::InvalidFunctionSignature("expected function name"),
1013        )?;
1014
1015        // fn House.verify(
1016        //    ^^^^^
1017        if is_type(&maybe_self_name.value) {
1018            let struct_name = maybe_self_name;
1019            // fn House.verify(
1020            //         ^
1021            tokens.bump_expected(ctx, TokenKind::Dot)?;
1022
1023            // fn House.verify(
1024            //          ^^^^^^
1025            let name = tokens.bump_ident(
1026                ctx,
1027                ErrorKind::InvalidFunctionSignature("expected function name"),
1028            )?;
1029
1030            Ok((
1031                name,
1032                FuncOrMethod::Method(CustomType {
1033                    module: ModulePath::Local,
1034                    name: struct_name.value,
1035                    span: struct_name.span,
1036                }),
1037            ))
1038        } else {
1039            // fn verify(
1040            //    ^^^^^^
1041
1042            // check that it is not shadowing a builtin
1043            let fn_name = maybe_self_name;
1044
1045            Ok((fn_name, FuncOrMethod::Function(ModulePath::Local)))
1046        }
1047    }

1050impl FunctionDef {
1051    pub fn is_main(&self) -> bool {
1052        self.sig.name.value == "main"
1053    }
1054
1055    pub fn parse_args(
1056        ctx: &mut ParserCtx,
1057        tokens: &mut Tokens,
1058        fn_kind: &FuncOrMethod,
1059    ) -> Result<Vec<FnArg>> {
1060        // (pub arg1: type1, arg2: type2)
1061        // ^
1062        tokens.bump_expected(ctx, TokenKind::LeftParen)?;
1063
1064        // (pub arg1: type1, arg2: type2)
1065        //   ^
1066        let mut args = vec![];
1067
1068        loop {
1069            // `pub arg1: type1`
1070            //   ^   ^
1071            let token = tokens.bump_err(
1072                ctx,
1073                ErrorKind::InvalidFunctionSignature("expected function arguments"),
1074            )?;
1075
1076            let (attribute, arg_name) = match token.kind {
1077                TokenKind::RightParen => break,
1078                // public input
1079                TokenKind::Keyword(Keyword::Pub) => {
1080                    let arg_name = Ident::parse(ctx, tokens)?;
1081                    (
1082                        Some(Attribute {
1083                            kind: AttributeKind::Pub,
1084                            span: token.span,
1085                        }),
1086                        arg_name,
1087                    )
1088                }
1089                // constant input
1090                TokenKind::Keyword(Keyword::Const) => {
1091                    let arg_name = Ident::parse(ctx, tokens)?;
1092                    (
1093                        Some(Attribute {
1094                            kind: AttributeKind::Const,
1095                            span: token.span,
1096                        }),
1097                        arg_name,
1098                    )
1099                }
1100                // private input
1101                TokenKind::Identifier(name) => (
1102                    None,
1103                    Ident {
1104                        value: name,
1105                        span: token.span,
1106                    },
1107                ),
1108                _ => {
1109                    return Err(ctx.error(
1110                        ErrorKind::InvalidFunctionSignature("expected identifier"),
1111                        token.span,
1112                    ));
1113                }
1114            };
1115
1116            // self takes no value
1117            let arg_typ = if arg_name.value == "self" {
1118                let self_name = match fn_kind {
1119                    FuncOrMethod::Function(_) => {
1120                        return Err(ctx.error(
1121                            ErrorKind::InvalidFunctionSignature(
1122                                "the `self` argument is only allowed in methods, not functions",
1123                            ),
1124                            arg_name.span,
1125                        ));
1126                    }
1127                    FuncOrMethod::Method(self_name) => self_name,
1128                };
1129
1130                if !args.is_empty() {
1131                    return Err(ctx.error(
1132                        ErrorKind::InvalidFunctionSignature("`self` must be the first argument"),
1133                        arg_name.span,
1134                    ));
1135                }
1136
1137                Ty {
1138                    kind: TyKind::Custom {
1139                        module: ModulePath::Local,
1140                        name: self_name.name.clone(),
1141                    },
1142                    span: self_name.span,
1143                }
1144            } else {
1145                // :
1146                tokens.bump_expected(ctx, TokenKind::Colon)?;
1147
1148                // type
1149                Ty::parse(ctx, tokens)?
1150            };
1151
1152            // , or )
1153            let separator = tokens.bump_err(
1154                ctx,
1155                ErrorKind::InvalidFunctionSignature("expected end of function or other argument"),
1156            )?;
1157
1158            let span = if let Some(attr) = &attribute {
1159                if &arg_name.value == "self" {
1160                    return Err(ctx.error(ErrorKind::SelfHasAttribute, arg_name.span));
1161                } else {
1162                    attr.span.merge_with(arg_typ.span)
1163                }
1164            } else {
1165                if &arg_name.value == "self" {
1166                    arg_name.span
1167                } else {
1168                    arg_name.span.merge_with(arg_typ.span)
1169                }
1170            };
1171
1172            let mut arg = FnArg {
1173                name: arg_name,
1174                typ: arg_typ,
1175                attribute,
1176                span,
1177            };
1178
1179            // if it is with const attribute, then converts it to a constant field.
1180            // this is because the parser doesn't know if a token has a corresponding attribute
1181            // until it has parsed the whole token.
1182            if arg.is_constant() {
1183                match &mut arg.typ.kind {
1184                    TyKind::Field { constant } => {
1185                        *constant = true;
1186                    }
1187                    _ => {
1188                        // for now we only allow constant fields
1189                        return Err(ctx.error(
1190                            ErrorKind::InvalidFunctionSignature(
1191                                "only allowed to have constant fields",
1192                            ),
1193                            arg.span,
1194                        ));
1195                    }
1196                }
1197            }
1198
1199            args.push(arg);
1200
1201            match separator.kind {
1202                // (pub arg1: type1, arg2: type2)
1203                //                 ^
1204                TokenKind::Comma => (),
1205                // (pub arg1: type1, arg2: type2)
1206                //                              ^
1207                TokenKind::RightParen => break,
1208                _ => {
1209                    return Err(ctx.error(
1210                        ErrorKind::InvalidFunctionSignature(
1211                            "expected end of function or other argument",
1212                        ),
1213                        separator.span,
1214                    ));
1215                }
1216            }
1217        }
1218
1219        Ok(args)
1220    }

1222    pub fn parse_fn_return_type(ctx: &mut ParserCtx, tokens: &mut Tokens) -> Result<Option<Ty>> {

1224            Some(Token {
1225                kind: TokenKind::RightArrow,
1226                ..
1227            }) => {
1228                tokens.bump(ctx);
1229
1230                let return_type = Ty::parse(ctx, tokens)?;
1231                Ok(Some(return_type))

1233            _ => Ok(None),

1237    pub fn parse_fn_body(ctx: &mut ParserCtx, tokens: &mut Tokens) -> Result<Vec<Stmt>> {
1238        let mut body = vec![];
1239
1240        // return empty body when the next token is `;` instead of `{`
1241        if matches!(
1242            tokens.peek(),
1243            Some(Token {
1244                kind: TokenKind::SemiColon,
1245                ..
1246            })
1247        ) {
1248            tokens.bump(ctx);
1249            return Ok(body);
1250        }
1251
1252        tokens.bump_expected(ctx, TokenKind::LeftCurlyBracket)?;
1253
1254        loop {
1255            // end of the function
1256            let next_token = tokens.peek();
1257            if matches!(
1258                next_token,
1259                Some(Token {
1260                    kind: TokenKind::RightCurlyBracket,
1261                    ..
1262                })
1263            ) {
1264                tokens.bump(ctx);
1265                break;
1266            }
1267
1268            // parse next statement
1269            let statement = Stmt::parse(ctx, tokens)?;
1270            body.push(statement);
1271        }
1272
1273        Ok(body)
1274    }

1276    /// Parse a function, without the `fn` keyword.
1277    pub fn parse(ctx: &mut ParserCtx, tokens: &mut Tokens) -> Result<Self> {
1278        // ghetto way of getting the span of the function: get the span of the first token (name), then try to get the span of the last token
1279        let mut span = tokens
1280            .peek()
1281            .ok_or_else(|| {
1282                ctx.error(
1283                    ErrorKind::InvalidFunctionSignature("expected function name"),
1284                    ctx.last_span(),
1285                )
1286            })?
1287            .span;
1288
1289        // parse signature
1290        let sig = FnSig::parse(ctx, tokens)?;
1291
1292        // make sure that it doesn't shadow a builtin
1293        if BUILTIN_FN_NAMES.contains(&sig.name.value.as_ref()) {
1294            return Err(ctx.error(
1295                ErrorKind::ShadowingBuiltIn(sig.name.value.clone()),
1296                sig.name.span,
1297            ));
1298        }
1299
1300        // parse body
1301        let body = Self::parse_fn_body(ctx, tokens)?;
1302
1303        // here's the last token, that is if the function is not empty (maybe we should disallow empty functions?)
1304
1305        if let Some(t) = body.last() {
1306            span = span.merge_with(t.span);
1307        } else {
1308            return Err(ctx.error(
1309                ErrorKind::InvalidFunctionSignature("expected function body"),
1310                ctx.last_span(),
1311            ));
1312        }
1313
1314        let func = Self {
1315            sig,
1316            body,
1317            span,
1318            is_hint: false,
1319        };
1320
1321        Ok(func)
1322    }

1324    /// Parse a hint function signature
1325    // todo: once builtin hints are deprecated merge this function to FunctionDef::parse (no need empty body for workaround)
1326    pub fn parse_hint(ctx: &mut ParserCtx, tokens: &mut Tokens) -> Result<Self> {
1327        // parse signature
1328        let sig = FnSig::parse(ctx, tokens)?;
1329        let mut span = sig.name.span;
1330
1331        // make sure that it doesn't shadow a builtin
1332        if BUILTIN_FN_NAMES.contains(&sig.name.value.as_ref()) {
1333            return Err(ctx.error(ErrorKind::ShadowingBuiltIn(sig.name.value.clone()), span));
1334        }
1335
1336        let body = Self::parse_fn_body(ctx, tokens)?;
1337
1338        if let Some(t) = body.last() {
1339            span = span.merge_with(t.span);
1340        }
1341
1342        let func = Self {
1343            sig,
1344            body,
1345            span,
1346            is_hint: true,
1347        };
1348
1349        Ok(func)
1350    }

1382// ## Statements
1383//
1384//~ statement ::=
1385//~     | "let" ident "=" expr ";"
1386//~     | expr ";"
1387//~     | "return" expr ";"
1388//~
1389//~ where an expression is allowed only if it is a function call that does not return a value.
1390//~
1391//~ Actually currently we don't implement it this way.
1392//~ We don't expect an expression to be a statement,
1393//~ but a well defined function call:
1394//~
1395//~ fn_call ::= path "(" [ expr { "," expr } ] ")"
1396//~ path ::= ident { "::" ident }
1397//~

1437impl Stmt {
1438    /// Parse the argument in the for loop
1439    /// if can be either a numerical range n..m or an array.
1440    fn parse_for_argument(ctx: &mut ParserCtx, tokens: &mut Tokens) -> Result<ForLoopArgument> {
1441        // for i in arg { ... }
1442        //          ^
1443        let start = Expr::parse(ctx, tokens)?;
1444
1445        // if the next token is DoubleDot, then it's a range
1446        // otherwise just return the start expression as the iterator argument
1447        match tokens.peek() {
1448            Some(Token {
1449                kind: TokenKind::DoubleDot,
1450                ..
1451            }) => {
1452                // for i in 0..5 { ... }
1453                //           ^^
1454                tokens.bump_expected(ctx, TokenKind::DoubleDot)?;
1455
1456                // for i in 0..5 { ... }
1457                //             ^
1458                let end = Expr::parse(ctx, tokens)?;
1459
1460                let start_span = start.span;
1461                let end_span = end.span;
1462
1463                let range = Range {
1464                    start,
1465                    end,
1466                    span: start_span.merge_with(end_span),
1467                };
1468
1469                Ok(ForLoopArgument::Range(range))
1470            }
1471            _ => Ok(ForLoopArgument::Iterator(Box::new(start))),
1472        }
1473    }

1475    /// Returns a list of statement parsed until seeing the end of a block (`}`).
1476    pub fn parse(ctx: &mut ParserCtx, tokens: &mut Tokens) -> Result<Self> {
1477        match tokens.peek() {
1478            None => Err(ctx.error(ErrorKind::InvalidStatement, ctx.last_span())),
1479            // assignment
1480            Some(Token {
1481                kind: TokenKind::Keyword(Keyword::Let),
1482                span,
1483            }) => {
1484                let mut span = span;
1485                tokens.bump(ctx);
1486
1487                // let mut x = 5;
1488                //     ^^^
1489
1490                let mutable = if matches!(
1491                    tokens.peek(),
1492                    Some(Token {
1493                        kind: TokenKind::Keyword(Keyword::Mut),
1494                        ..
1495                    })
1496                ) {
1497                    tokens.bump(ctx);
1498                    true
1499                } else {
1500                    false
1501                };
1502
1503                // let mut x = 5;
1504                //         ^
1505                let lhs = Ident::parse(ctx, tokens)?;
1506
1507                // let mut x = 5;
1508                //           ^
1509                tokens.bump_expected(ctx, TokenKind::Equal)?;
1510
1511                // let mut x = 5;
1512                //             ^
1513                let rhs = Box::new(Expr::parse(ctx, tokens)?);
1514                span = span.merge_with(rhs.span);
1515
1516                // let mut x = 5;
1517                //              ^
1518                tokens.bump_expected(ctx, TokenKind::SemiColon)?;
1519
1520                //
1521                Ok(Stmt {
1522                    kind: StmtKind::Assign { mutable, lhs, rhs },
1523                    span,
1524                })
1525            }
1526
1527            // for loop, can be either a range or an iterator
1528            Some(Token {
1529                kind: TokenKind::Keyword(Keyword::For),
1530                span,
1531            }) => {
1532                tokens.bump(ctx);
1533
1534                // for i in arg { ... }
1535                //     ^
1536                let var = Ident::parse(ctx, tokens)?;
1537
1538                // for i in arg { ... }
1539                //       ^^
1540                tokens.bump_expected(ctx, TokenKind::Keyword(Keyword::In))?;
1541
1542                let argument = Self::parse_for_argument(ctx, tokens)?;
1543
1544                // for i in arg { ... }
1545                //               ^
1546                tokens.bump_expected(ctx, TokenKind::LeftCurlyBracket)?;
1547
1548                // for i in arg { ... }
1549                //                 ^^^
1550                let mut body = vec![];
1551
1552                loop {
1553                    // for i in 0..5 { ... }
1554                    //                     ^
1555                    let next_token = tokens.peek();
1556                    if matches!(
1557                        next_token,
1558                        Some(Token {
1559                            kind: TokenKind::RightCurlyBracket,
1560                            ..
1561                        })
1562                    ) {
1563                        tokens.bump(ctx);
1564                        break;
1565                    }
1566
1567                    // parse next statement
1568                    // TODO: should we prevent `return` here?
1569                    // TODO: in general, do we prevent early returns atm?
1570                    let statement = Stmt::parse(ctx, tokens)?;
1571                    body.push(statement);
1572                }
1573                Ok(Stmt {
1574                    kind: StmtKind::ForLoop {
1575                        var,
1576                        argument,
1577                        body,
1578                    },
1579                    span,
1580                })
1581            }
1582
1583            // if/else
1584            Some(Token {
1585                kind: TokenKind::Keyword(Keyword::If),
1586                span,
1587            }) => {
1588                // TODO: wait, this should be implemented as an expression! not a statement
1589                Err(Error::new("parse", ErrorKind::UnexpectedError("if statements are not implemented yet. Use if expressions instead (e.g. `x = if cond {{ 1 }} else {{ 2 }};`)"), span))?
1590            }
1591
1592            // return
1593            Some(Token {
1594                kind: TokenKind::Keyword(Keyword::Return),
1595                span,
1596            }) => {
1597                tokens.bump(ctx);
1598
1599                // return xx;
1600                //        ^^
1601                let expr = Expr::parse(ctx, tokens)?;
1602
1603                // return xx;
1604                //          ^
1605                let end_token = tokens.bump_expected(ctx, TokenKind::SemiColon)?;
1606
1607                Ok(Stmt {
1608                    kind: StmtKind::Return(Box::new(expr)),
1609                    span: span.merge_with(end_token.span),
1610                })
1611            }
1612
1613            // comment
1614            Some(Token {
1615                kind: TokenKind::Comment(c),
1616                span,
1617            }) => {
1618                tokens.bump(ctx);
1619                Ok(Stmt {
1620                    kind: StmtKind::Comment(c),
1621                    span,
1622                })
1623            }
1624
1625            // statement expression (like function call)
1626            _ => {
1627                let expr = Expr::parse(ctx, tokens)?;
1628                let span = expr.span;
1629
1630                tokens.bump_expected(ctx, TokenKind::SemiColon)?;
1631
1632                Ok(Stmt {
1633                    kind: StmtKind::Expr(Box::new(expr)),
1634                    span,
1635                })
1636            }
1637        }
1638    }
1639}

1646#[derive(Debug, Serialize)]
1647
1648/// Things you can have in a scope (including the root scope).
1649pub struct Root<F>
1650where
1651    F: Field,
1652{
1653    #[serde(bound(serialize = "RootKind<F>: Serialize"))]
1654    pub kind: RootKind<F>,

1657
1658#[derive(Clone, Debug, Serialize, Deserialize)]
1659pub struct UsePath {
1660    pub module: Ident,
1661    pub submodule: Ident,
1662    pub span: Span,
1663}
1664
1665impl From<&UsePath> for UserRepo {
1666    fn from(path: &UsePath) -> Self {
1667        UserRepo {
1668            user: path.module.value.clone(),
1669            repo: path.submodule.value.clone(),
1670        }
1671    }
1672}
1673
1674impl Display for UsePath {
1675    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
1676        write!(f, "{}::{}", self.module.value, self.submodule.value)
1677    }
1678}
1679
1680impl UsePath {
1681    pub fn parse(ctx: &mut ParserCtx, tokens: &mut Tokens) -> Result<Self> {
1682        let module = tokens.bump_ident(
1683            ctx,
1684            ErrorKind::InvalidPath("wrong path: expected a module (TODO: better error"),
1685        )?;
1686        let span = module.span;
1687
1688        tokens.bump_expected(ctx, TokenKind::DoubleColon)?; // ::
1689
1690        let submodule = tokens.bump_ident(
1691            ctx,
1692            ErrorKind::InvalidPath(
1693                "wrong path: expected a submodule after `::` (TODO: better error",
1694            ),
1695        )?;
1696
1697        let span = span.merge_with(submodule.span);
1698        Ok(UsePath {
1699            module,
1700            submodule,
1701            span,
1702        })
1703    }
1704}

1706#[derive(Debug, Serialize)]
1707pub enum RootKind<F: Field> {
1708    Use(UsePath),
1709    FunctionDef(FunctionDef),
1710    Comment(String),
1711    StructDef(StructDef),
1712    #[serde(bound(serialize = "ConstDef<F>: Serialize"))]
1713    ConstDef(ConstDef<F>),
1714}

1717// Const
1718//
1719
1720#[serde_as]
1721#[derive(Debug, Serialize)]
1722pub struct ConstDef<F>
1723where
1724    F: Field,
1725{
1726    pub module: ModulePath, // name resolution
1727    pub name: Ident,
1728    #[serde_as(as = "crate::serialization::SerdeAs")]
1729    pub value: F,
1730    pub span: Span,
1731}
1732
1733impl<F: Field + FromStr> ConstDef<F> {
1734    pub fn parse(ctx: &mut ParserCtx, tokens: &mut Tokens) -> Result<Self> {
1735        // const foo = 42;
1736        //       ^^^
1737        let name = Ident::parse(ctx, tokens)?;
1738
1739        // const foo = 42;
1740        //           ^
1741        tokens.bump_expected(ctx, TokenKind::Equal)?;
1742
1743        // const foo = 42;
1744        //             ^^
1745        let value = Expr::parse(ctx, tokens)?;
1746        let value = match &value.kind {
1747            ExprKind::BigUInt(n) => n
1748                .to_string()
1749                .parse()
1750                .map_err(|_e| ctx.error(ErrorKind::InvalidField(n.to_string()), value.span))?,
1751            _ => {
1752                return Err(ctx.error(ErrorKind::InvalidConstType, value.span));
1753            }
1754        };
1755
1756        // const foo = 42;
1757        //               ^
1758        tokens.bump_expected(ctx, TokenKind::SemiColon)?;
1759
1760        //
1761        let span = name.span;
1762        Ok(ConstDef {
1763            module: ModulePath::Local,
1764            name,
1765            value,
1766            span,
1767        })
1768    }
