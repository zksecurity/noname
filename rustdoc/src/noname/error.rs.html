14/// An error in noname.
15#[derive(Diagnostic, Debug, Error)]
16#[error("Looks like something went wrong in {label}")]
17pub struct Error {
18    /// A hint as to where the error happened (e.g. type-checker, parser, lexer, etc.)
19    pub label: &'static str,
20
21    /// The type of error.
22    #[help]
23    pub kind: ErrorKind,
24
25    /// Indicate where the error occurred in the source code.
26    #[label("here")]
27    pub span: Span,
28}
29
30impl Error {
31    /// Creates a new [Error] from an [ErrorKind].
32    pub fn new(label: &'static str, kind: ErrorKind, span: Span) -> Self {
33        Self { label, kind, span }
34    }
35}
