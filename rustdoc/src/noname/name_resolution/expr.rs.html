1use crate::{
2    cli::packages::UserRepo,
3    error::Result,
4    parser::{types::ModulePath, CustomType, Expr, ExprKind},
5    stdlib::builtins::{BUILTIN_FN_NAMES, QUALIFIED_BUILTINS},
6};
7
8use super::context::NameResCtx;
9
10impl NameResCtx {
11    pub(crate) fn resolve_expr(&self, expr: &mut Expr) -> Result<()> {
12        let Expr {
13            node_id: _,
14            kind,
15            span: _,
16        } = expr;
17
18        match kind {
19            ExprKind::FnCall {
20                module,
21                fn_name,
22                args,
23                unsafe_attr: _,
24            } => {
25                if matches!(module, ModulePath::Local)
26                    && BUILTIN_FN_NAMES.contains(&fn_name.value.as_str())
27                {
28                    // if it's a builtin, use `std::builtin`
29                    *module = ModulePath::Absolute(UserRepo::new(QUALIFIED_BUILTINS));
30                } else {
31                    self.resolve(module, false)?;
32                }
33
34                for arg in args {
35                    self.resolve_expr(arg)?;
36                }
37            }
38            ExprKind::MethodCall {
39                lhs,
40                method_name: _,
41                args,
42            } => {
43                self.resolve_expr(lhs)?;
44                for arg in args {
45                    self.resolve_expr(arg)?;
46                }
47            }
48            ExprKind::Assignment { lhs, rhs } => {
49                self.resolve_expr(lhs)?;
50                self.resolve_expr(rhs)?;
51            }
52            ExprKind::FieldAccess { lhs, rhs: _ } => {
53                self.resolve_expr(lhs)?;
54            }
55            ExprKind::BinaryOp {
56                op: _,
57                lhs,
58                rhs,
59                protected: _,
60            } => {
61                self.resolve_expr(lhs)?;
62                self.resolve_expr(rhs)?;
63            }
64            ExprKind::Negated(expr) => {
65                self.resolve_expr(expr)?;
66            }
67            ExprKind::Not(expr) => {
68                self.resolve_expr(expr)?;
69            }
70            ExprKind::BigUInt(_) => {}
71            ExprKind::Variable { module, name: _ } => {
72                self.resolve(module, false)?;
73            }
74            ExprKind::ArrayOrTupleAccess { container, idx } => {
75                self.resolve_expr(container)?;
76                self.resolve_expr(idx)?;
77            }
78            ExprKind::ArrayDeclaration(items) => {
79                for expr in items {
80                    self.resolve_expr(expr)?;
81                }
82            }
83            ExprKind::RepeatedArrayInit { item, size } => {
84                self.resolve_expr(item)?;
85                self.resolve_expr(size)?;
86            }
87            ExprKind::CustomTypeDeclaration {
88                custom: struct_name,
89                fields,
90            } => {
91                let CustomType {
92                    module,
93                    name: _,
94                    span: _,
95                } = struct_name;
96                self.resolve(module, true)?;
97                for (_field_name, field_value) in fields {
98                    self.resolve_expr(field_value)?;
99                }
100            }
101            ExprKind::Bool(_) => {}
102            ExprKind::StringLiteral(_) => {}
103            ExprKind::IfElse { cond, then_, else_ } => {
104                self.resolve_expr(cond)?;
105                self.resolve_expr(then_)?;
106                self.resolve_expr(else_)?;
107            }
108            ExprKind::TupleDeclaration(items) => {
109                for expr in items {
110                    self.resolve_expr(expr)?;
111                }
112            }
113        };
114
115        Ok(())
116    }
117}
