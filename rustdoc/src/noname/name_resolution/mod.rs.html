16pub struct NAST<B>
17where
18    B: Backend,

20    #[serde(bound(serialize = "AST<B>: Serialize"))]
21    pub ast: AST<B>,
22}

24impl<B: Backend> NAST<B> {
25    fn new(ast: AST<B>) -> Self {
26        Self { ast }
27    }
28
29    pub fn resolve_modules(this_module: Option<UserRepo>, mut ast: AST<B>) -> Result<NAST<B>> {
30        let mut ctx = NameResCtx::new(this_module);
31
32        // create a map of the imported modules (and how they are aliases)
33        let mut abort = None;
34        for root in &ast.0 {
35            match &root.kind {
36                // `use user::repo;`
37                RootKind::Use(path) => {
38                    // important: no struct or function definition can appear before a use declaration
39                    if let Some(span) = abort {
40                        Err(Error::new(
41                            "type-checker",
42                            ErrorKind::OrderOfUseDeclaration,
43                            span,
44                        ))?;
45                    }
46
47                    // insert and detect duplicates
48                    let dedup = ctx
49                        .modules
50                        .insert(path.submodule.value.clone(), path.clone());
51                    if let Some(_) = dedup {
52                        Err(ctx.error(
53                            ErrorKind::DuplicateModule(path.submodule.value.clone()),
54                            path.submodule.span,
55                        ))?
56                    }
57                }
58                RootKind::FunctionDef(FunctionDef { span, .. })
59                | RootKind::StructDef(StructDef { span, .. })
60                | RootKind::ConstDef(ConstDef { span, .. }) => abort = Some(*span),
61                RootKind::Comment(_) => (),
62            }
63        }
64
65        // now go through the AST and mutate any module to its fully-qualified path
66        for root in &mut ast.0 {
67            match &mut root.kind {
68                RootKind::FunctionDef(f) => ctx.resolve_fn_def(f)?,
69                RootKind::StructDef(s) => ctx.resolve_struct_def(s)?,
70                RootKind::ConstDef(c) => ctx.resolve_const_def(c)?,
71                RootKind::Use(_) | RootKind::Comment(_) => (),
72            }
73        }
74
75        Ok(NAST::new(ast))
76    }
77}

93    use user::repo;
94
95    const some_cst = 0;
96
97    struct Thing {
98        bb: Field,
99    }
100
101    fn some_func(xx: [Field; 3], yy: Field) -> [Field; 3] {
102        let aa = Thing { bb: yy };
103        return repo::thing(xx, aa);
104    }
105"#;

107    #[test]
108    fn test_name_res() {
109        let tokens = Token::parse(0, CODE).unwrap();
110        let (ast, _node_id) = AST::<KimchiVesta>::parse(0, tokens, 0).unwrap();
111        let nast = NAST::resolve_modules(None, ast).unwrap();
112
113        // find constant declaration
114        let mut roots = nast
115            .ast
116            .0
117            .iter()
118            .skip_while(|r| !matches!(r.kind, RootKind::ConstDef(_)));
119        let cst = match &roots.next().unwrap().kind {
120            RootKind::ConstDef(c) => c,
121            _ => panic!("expected const def"),
122        };
123        assert!(matches!(cst.module, ModulePath::Local));
124
125        // struct definition
126        let struct_def = match &roots.next().unwrap().kind {
127            RootKind::StructDef(d) => d,
128            _ => panic!("expected const def"),
129        };
130        assert!(matches!(struct_def.module, ModulePath::Local));
131
132        // return statement
133        let fn_def = match &roots.next().unwrap().kind {
134            RootKind::FunctionDef(d) => d,
135            _ => panic!("expected const def"),
136        };
137        let fn_call = match &fn_def.body[1].kind {
138            StmtKind::Return(e) => e,
139            _ => panic!("expected assignment"),
140        };
141        let module = match &fn_call.kind {
142            ExprKind::FnCall { module, .. } => module,
143            _ => panic!("expected struct"),
144        };
145
146        match module {
147            ModulePath::Absolute(u) if u == &UserRepo::new("user/repo") => (),
148            _ => panic!("expected absolute module path"),
149        };
150    }
