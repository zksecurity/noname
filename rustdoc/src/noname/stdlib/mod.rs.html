63type FnInfoType<B: Backend> = fn(
64    &mut CircuitWriter<B>,
65    &GenericParameters,
66    &[VarInfo<B::Field, B::Var>],
67    Span,
68) -> Result<Option<Var<B::Field, B::Var>>>;
69
70trait Module {
71    /// e.g. "crypto"
72    const MODULE: &'static str;
73
74    fn get_fns<B: Backend>() -> Vec<(&'static str, FnInfoType<B>, bool)>;
75
76    fn get_parsed_fns<B: Backend>() -> Vec<FnInfo<B>> {
77        let fns = Self::get_fns();
78        let mut res = Vec::with_capacity(fns.len());
79        for (code, fn_handle, ignore_arg_types) in fns {
80            let ctx = &mut ParserCtx::default();
81            // TODO: we should try to point to real noname files here (not 0)
82            let mut tokens = Token::parse(0, code).unwrap();
83            let sig = FnSig::parse(ctx, &mut tokens).unwrap();
84            res.push(FnInfo {
85                kind: FnKind::BuiltIn(sig, fn_handle, ignore_arg_types),
86                is_hint: false,
87                span: Span::default(),
88            });
89        }
90        res
91    }
92}

94pub fn init_stdlib_dep<B: Backend>(
95    sources: &mut Sources,
96    tast: &mut TypeChecker<B>,
97    node_id: usize,
98    path_prefix: &str,
99    server_mode: &mut Option<crate::server::ServerShim>,
100) -> usize {
101    // list the stdlib dependency in order
102    let libs = vec!["bits", "comparator", "multiplexer", "mimc", "int"];
103
104    let mut node_id = node_id;
105
106    for lib in libs {
107        let module = UserRepo::new(&format!("std/{}", lib));
108        let prefix_stdlib = Path::new(path_prefix);
109        let code = std::fs::read_to_string(prefix_stdlib.join(format!("{lib}/lib.no"))).unwrap();
110        node_id = typecheck_next_file(
111            tast,
112            Some(module),
113            sources,
114            lib.to_string(),
115            code,
116            node_id,
117            server_mode,
118        )
119        .unwrap();
120    }
121
122    node_id
123}
